41  /**
42   * A simple embedded server that lets us run directly within Eclipse with added group validation
43   * 
44   * Browse to http://localhost:8080/ to test.
45   */
46  public class StartEmbeddedJettyValidateNTLMGroup {
47  
48      /** The logger. */
49      private static final Logger LOGGER = LoggerFactory.getLogger(StartEmbeddedJettyValidateNTLMGroup.class);
50  
51      /**
52       * The main method.
53       *
54       * @param args
55       *            the arguments
56       */
57      public static void main(final String args[]) {
58          System.setProperty(SimpleLogger.DEFAULT_LOG_LEVEL_KEY, "TRACE");
59  
60          final Server server = new Server(8080);
61  
62          final ServletContextHandler context = new ServletContextHandler(ServletContextHandler.SESSIONS);
63          context.setContextPath("/");
64  
65          final ServletHandler handler = new ServletHandler();
66          final ServletHolder sh = new ServletHolder(new InfoServlet());
67          context.addServlet(sh, "/*");
68  
69          final FilterHolder fh = handler.addFilterWithMapping(NegotiateSecurityFilter.class, "/*",
70                  EnumSet.of(DispatcherType.REQUEST));
71          StartEmbeddedJettyValidateNTLMGroup.setFilterParams(fh);
72          context.addFilter(fh, "/*", EnumSet.of(DispatcherType.REQUEST));
73  
74          context.setHandler(handler);
75          server.setHandler(context);
76  
77          try {
78              server.start();
79          } catch (final Exception e) {
80              StartEmbeddedJettyValidateNTLMGroup.LOGGER.trace("", e);
81          }
82      }
83  
84      /**
85       * Sets the filter params.
86       *
87       * @param fh
88       *            the new filter params
89       */
90      private static void setFilterParams(final FilterHolder fh) {
91          fh.setInitParameter("principalFormat", "fqn");
92          fh.setInitParameter("roleFormat", "both");
93  
94          fh.setInitParameter("allowGuestLogin", "true");
95          fh.setInitParameter("impersonate", "true");
96  
97          fh.setInitParameter("securityFilterProviders",
98                  "waffle.servlet.spi.NegotiateSecurityFilterProvider waffle.servlet.spi.BasicSecurityFilterProvider");
99          fh.setInitParameter("waffle.servlet.spi.NegotiateSecurityFilterProvider/protocols", "Negotiate NTLM");
100 
101         fh.setInitParameter("waffle.servlet.spi.BasicSecurityFilterProvider/realm", "SecureServiceRunner");
102     }
103 
104     /**
105      * The Class InfoServlet.
106      */
107     public static class InfoServlet extends HttpServlet {
108 
109         /** The Constant serialVersionUID. */
110         private static final long serialVersionUID = 1L;
111 
112         /** The authorised groups. */
113         private static final List<String> authorisedGroups = Arrays.asList("NTGroup1", "NTGroup2");
114 
115         @Override
116         public void doGet(final HttpServletRequest request, final HttpServletResponse response)
117                 throws ServletException, IOException {
118             response.setContentType("text/html");
119             response.setStatus(HttpServletResponse.SC_OK);
120 
121             final boolean isUserAuthorised = this.isUserAuthorised(request, InfoServlet.authorisedGroups);
122             if (isUserAuthorised) {
123                 response.getWriter().println("User is authorised");
124             } else {
125                 response.getWriter().println("User is not authorised");
126             }
127         }
128 
129         /**
130          * Checks if is user authorised.
131          *
132          * @param request
133          *            the request
134          * @param authorizedGroups
135          *            the authorized groups
136          * @return true, if is user authorised
137          */
138         private boolean isUserAuthorised(final HttpServletRequest request, final List<String> authorizedGroups) {
139             final List<String> usersGroups = this.getUsersGroups(request);
140 
141             final boolean noOverlappingGroups = Collections.disjoint(authorizedGroups, usersGroups);
142             return !noOverlappingGroups;
143         }
144 
145         /**
146          * Gets the users groups.
147          *
148          * @param request
149          *            the request
150          * @return the users groups
151          */
152         private List<String> getUsersGroups(final HttpServletRequest request) {
153             final List<String> result = new ArrayList<>();
154             final Principal principal = request.getUserPrincipal();
155             if (principal instanceof WindowsPrincipal) {
156                 String groupName;
157                 final WindowsPrincipal windowsPrincipal = (WindowsPrincipal) principal;
158                 for (final WindowsAccount account : windowsPrincipal.getGroups().values()) {
159                     groupName = this.getGroupName(account.getDomain(), account.getFqn());
160                     result.add(groupName);
161                 }
162             }
163             return result;
164         }
165 
166         /**
167          * Gets the group name.
168          *
169          * @param domain
170          *            the domain
171          * @param groupString
172          *            the group string
173          * @return the group name
174          */
175         private String getGroupName(final String domain, final String groupString) {
176             if (domain == null || groupString == null) {
177                 return "";
178             }
179             return groupString.split(domain)[1].substring(1);
180         }
181     }
