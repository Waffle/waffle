22  public class NtlmMessageTests {
23  
24      @Test
25      public void testIsNtlmMessage() {
26          Assert.assertFalse(NtlmMessage.isNtlmMessage(null));
27          final byte[] ntlmSignature = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00 };
28          Assert.assertTrue(NtlmMessage.isNtlmMessage(ntlmSignature));
29          final byte[] shortMessage = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50 };
30          Assert.assertFalse(NtlmMessage.isNtlmMessage(shortMessage));
31          final byte[] longMessage = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x00 };
32          Assert.assertTrue(NtlmMessage.isNtlmMessage(longMessage));
33          final byte[] badMessage = { 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00 };
34          Assert.assertFalse(NtlmMessage.isNtlmMessage(badMessage));
35      }
36  
37      @Test
38      public void testGetNtlmMessageType() {
39          final byte[] ntlmMessageType1 = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x01, 0x00, 0x00, 0x00, 0x02,
40                  0x02, 0x00, 0x00 };
41          Assert.assertEquals(1, NtlmMessage.getMessageType(ntlmMessageType1));
42          final byte[] ntlmMessageType2 = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x02, 0x00, 0x00, 0x00, 0x00,
43                  0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x02, 0x02, 0x00, 0x00, 0x01, 0x23, 0x45, 0x67 };
44          Assert.assertEquals(2, NtlmMessage.getMessageType(ntlmMessageType2));
45          final byte[] ntlmMessageType3 = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x03, 0x00, 0x00, 0x00, 0x00,
46                  0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x02, 0x02, 0x00, 0x00, 0x01, 0x23, 0x45, 0x67 };
47          Assert.assertEquals(3, NtlmMessage.getMessageType(ntlmMessageType3));
48      }
