24  package waffle.util;
25  
26  import org.junit.jupiter.api.Assertions;
27  import org.junit.jupiter.api.Test;
28  
29  /**
30   * The Class SPNegoMessageTest.
31   */
32  class SPNegoMessageTest {
33  
34      // Different SPNEGO messages. For details and specification,
35      // see https://msdn.microsoft.com/en-us/library/ms995330.aspx
36  
37      /** The Constant NEG_TOKEN_INIT_OK. */
38      private static final byte[] NEG_TOKEN_INIT_OK = { 0x60, 0x76, 0x06, 0x06, 0x2B, 0x06, 0x01, 0x05, 0x05, 0x02 };
39  
40      /** The Constant NEG_TOKEN_INIT_TOO_SHORT. */
41      private static final byte[] NEG_TOKEN_INIT_TOO_SHORT = { 0x60, 0x76, 0x06, 0x06, 0x2B, 0x06, 0x01, 0x05, 0x05 };
42  
43      /** The Constant NEG_TOKEN_ARG_OK. */
44      private static final byte[] NEG_TOKEN_ARG_OK = { (byte) 0xA1, 0x33, 0x30, 0x31, 0x0, 0x03, 0x0A, 0x01, 0x01, 0x0,
45              0x2A, 0x04, 0x28, 0x4E, 0x54, 0x4C, 0x4D, 0x53, 0x53, 0x50, 0x00, 0x01, 0x00, 0x00, 0x00, 0x0, 0x0, 0x08,
46              0x0, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x06,
47              0x01, 0x0, 0x1D, 0x00, 0x00, 0x00, 0x0F };
48  
49      /** The Constant NEG_TOKEN_ARG_TOO_SHORT. */
50      private static final byte[] NEG_TOKEN_ARG_TOO_SHORT = { (byte) 0xA1, 0x33, 0x30, 0x31, 0x0, 0x03 };
51  
52      /** The Constant BAD_MESSAGE. */
53      private static final byte[] BAD_MESSAGE = { 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00 };
54  
55      /**
56       * Test is neg token init.
57       */
58      @Test
59      void testIsNegTokenInit() {
60          Assertions.assertTrue(SPNegoMessage.isNegTokenInit(SPNegoMessageTest.NEG_TOKEN_INIT_OK));
61          Assertions.assertFalse(SPNegoMessage.isNegTokenInit(SPNegoMessageTest.NEG_TOKEN_INIT_TOO_SHORT));
62          Assertions.assertFalse(SPNegoMessage.isNegTokenInit(SPNegoMessageTest.BAD_MESSAGE));
63      }
64  
65      /**
66       * Test is neg token arg.
67       */
68      @Test
69      void testIsNegTokenArg() {
70          Assertions.assertTrue(SPNegoMessage.isNegTokenArg(SPNegoMessageTest.NEG_TOKEN_ARG_OK));
71          Assertions.assertFalse(SPNegoMessage.isNegTokenArg(SPNegoMessageTest.NEG_TOKEN_ARG_TOO_SHORT));
72          Assertions.assertFalse(SPNegoMessage.isNegTokenArg(SPNegoMessageTest.BAD_MESSAGE));
73      }
74  }
