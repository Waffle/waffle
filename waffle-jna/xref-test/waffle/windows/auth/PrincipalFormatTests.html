17  /**
18   * The Class PrincipalFormatTests.

22  public class PrincipalFormatTests {
23  
24      /**
25       * Test known.
26       */
27      @Test
28      public void testKnown() {
29          Assertions.assertEquals(PrincipalFormat.FQN, PrincipalFormat.valueOf("FQN"));
30          Assertions.assertEquals(PrincipalFormat.SID, PrincipalFormat.valueOf("SID"));
31          Assertions.assertEquals(PrincipalFormat.BOTH, PrincipalFormat.valueOf("BOTH"));
32          Assertions.assertEquals(PrincipalFormat.NONE, PrincipalFormat.valueOf("NONE"));
33          Assertions.assertEquals(4, PrincipalFormat.values().length);
34      }
35  
36      /**
37       * Test unknown.
38       */
39      @Test
40      public void testUnknown() {
41          Assertions.assertThrows(RuntimeException.class, () -> {
42              PrincipalFormat.valueOf("garbage");
43          });
44      }
