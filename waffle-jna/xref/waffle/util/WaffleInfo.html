53  /**
54   * A Utility class to read system info to help troubleshoot WAFFLE system configuration.
55   *
56   * <pre>
57   * This utility class collects system information and returns it as an XML document.
58   * </pre>
59   *
60   * From the command line, you can write the info to stdout using:
61   *
62   * <pre>
63   * <code>
64   *   java -cp "jna.jar;waffle-core.jar;waffle-api.jar;jna-platform.jar;guava-21.0.jar" waffle.util.WaffleInfo
65   * </code>
66   * </pre>
67   *
68   * To show this information in a browser, run:
69   *
70   * <pre>
71   * <code>
72   *   java -cp "..." waffle.util.WaffleInfo -show
73   * </code>
74   * </pre>
75   *
76   * To lookup account names and return any listed info, run:
77   *
78   * <pre>
79   * <code>
80   *   java -cp "..." waffle.util.WaffleInfo -lookup AccountName
81   * </code>
82   * </pre>
83   *
84   */
85  public class WaffleInfo {

90      /**
91       * Get a Document with basic system information
92       *
93       * This uses the builtin javax.xml package even though the API is quite verbose
94       *
95       * @return Document with waffle info.
96       *
97       * @throws ParserConfigurationException
98       *             when getting new document builder.
99       */
100     public Document getWaffleInfo() throws ParserConfigurationException {
101         final DocumentBuilderFactory df = DocumentBuilderFactory.newInstance();
102         df.setAttribute(XMLConstants.ACCESS_EXTERNAL_DTD, "");
103         df.setAttribute(XMLConstants.ACCESS_EXTERNAL_SCHEMA, "");
104         df.setExpandEntityReferences(false);
105 
106         final Document doc = df.newDocumentBuilder().newDocument();
107 
108         // create the root element and add it to the document
109         final Element root = doc.createElement("waffle");
110 
111         // Add Version Information as attributes
112         String version = WaffleInfo.class.getPackage().getImplementationVersion();
113         if (version != null) {
114             root.setAttribute("version", version);
115         }
116         version = Platform.class.getPackage().getImplementationVersion();
117         if (version != null) {
118             root.setAttribute("jna", version);
119         }
120         version = WindowUtils.class.getPackage().getImplementationVersion();
121         if (version != null) {
122             root.setAttribute("jna-platform", version);
123         }
124 
125         doc.appendChild(root);
126         root.appendChild(this.getAuthProviderInfo(doc));
127 
128         return doc;
129     }

131     /**
132      * Gets the auth provider info.
133      *
134      * @param doc
135      *            the doc
136      * @return the auth provider info
137      */
138     protected Element getAuthProviderInfo(final Document doc) {
139         final IWindowsAuthProvider auth = new WindowsAuthProviderImpl();
140 
141         final Element node = doc.createElement("auth");
142         node.setAttribute("class", auth.getClass().getName());
143 
144         // Current User
145         Element child = doc.createElement("currentUser");
146         node.appendChild(child);
147 
148         final String currentUsername = WindowsAccountImpl.getCurrentUsername();
149         this.addAccountInfo(doc, child, new WindowsAccountImpl(currentUsername));
150 
151         // Computer
152         child = doc.createElement("computer");
153         node.appendChild(child);
154 
155         final IWindowsComputer c = auth.getCurrentComputer();
156         Element value = doc.createElement("computerName");
157         value.setTextContent(c.getComputerName());
158         child.appendChild(value);
159 
160         value = doc.createElement("memberOf");
161         value.setTextContent(c.getMemberOf());
162         child.appendChild(value);
163 
164         value = doc.createElement("joinStatus");
165         value.setTextContent(c.getJoinStatus());
166         child.appendChild(value);
167 
168         value = doc.createElement("groups");
169         Element g;
170         for (final String s : c.getGroups()) {
171             g = doc.createElement("group");
172             g.setTextContent(s);
173             value.appendChild(g);
174         }
175         child.appendChild(value);
176 
177         // Only Show Domains if we are in a Domain
178         if (Netapi32Util.getJoinStatus() == LMJoin.NETSETUP_JOIN_STATUS.NetSetupDomainName) {
179             child = doc.createElement("domains");
180             node.appendChild(child);
181 
182             Element d;
183             for (final IWindowsDomain domain : auth.getDomains()) {
184                 d = doc.createElement("domain");
185                 node.appendChild(d);
186 
187                 value = doc.createElement("FQN");
188                 value.setTextContent(domain.getFqn());
189                 child.appendChild(value);
190 
191                 value = doc.createElement("TrustTypeString");
192                 value.setTextContent(domain.getTrustTypeString());
193                 child.appendChild(value);
194 
195                 value = doc.createElement("TrustDirectionString");
196                 value.setTextContent(domain.getTrustDirectionString());
197                 child.appendChild(value);
198             }
199         }
200         return node;
201     }

203     /**
204      * Adds the account info.
205      *
206      * @param doc
207      *            the doc
208      * @param node
209      *            the node
210      * @param account
211      *            the account
212      */
213     protected void addAccountInfo(final Document doc, final Element node, final IWindowsAccount account) {
214         Element value = doc.createElement("Name");
215         value.setTextContent(account.getName());
216         node.appendChild(value);
217 
218         value = doc.createElement("FQN");
219         value.setTextContent(account.getFqn());
220         node.appendChild(value);
221 
222         value = doc.createElement("Domain");
223         value.setTextContent(account.getDomain());
224         node.appendChild(value);
225 
226         value = doc.createElement("SID");
227         value.setTextContent(account.getSidString());
228         node.appendChild(value);
229     }

231     /**
232      * Gets the lookup info.
233      *
234      * @param doc
235      *            the doc
236      * @param lookup
237      *            the lookup
238      * @return the lookup info
239      */
240     public Element getLookupInfo(final Document doc, final String lookup) {
241         final IWindowsAuthProvider auth = new WindowsAuthProviderImpl();
242         final Element node = doc.createElement("lookup");
243         node.setAttribute("name", lookup);
244         try {
245             this.addAccountInfo(doc, node, auth.lookupAccount(lookup));
246         } catch (final Win32Exception e) {
247             node.appendChild(WaffleInfo.getException(doc, e));
248         }
249         return node;
250     }

252     /**
253      * Gets the exception.
254      *
255      * @param doc
256      *            the doc
257      * @param t
258      *            the t
259      * @return the exception
260      */
261     public static Element getException(final Document doc, final Exception t) {
262         final Element node = doc.createElement("exception");
263         node.setAttribute("class", t.getClass().getName());
264 
265         Element value = doc.createElement("message");
266         if (t.getMessage() != null) {
267             value.setTextContent(t.getMessage());
268             node.appendChild(value);
269         }
270 
271         value = doc.createElement("trace");
272         value.setTextContent(Arrays.toString(t.getStackTrace()));
273         node.appendChild(value);
274         return node;
275     }

277     /**
278      * To pretty xml.
279      *
280      * @param doc
281      *            the doc
282      * @return the string
283      * @throws TransformerException
284      *             the transformer exception
285      */
286     public static String toPrettyXML(final Document doc) throws TransformerException {
287         // set up a transformer
288         final TransformerFactory transfac = TransformerFactory.newInstance();
289         transfac.setAttribute(XMLConstants.ACCESS_EXTERNAL_DTD, "");
290         transfac.setAttribute(XMLConstants.ACCESS_EXTERNAL_STYLESHEET, "");
291 
292         final Transformer trans = transfac.newTransformer();
293         trans.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
294         trans.setOutputProperty(OutputKeys.INDENT, "yes");
295 
296         // create string from xml tree
297         final StringWriter sw = new StringWriter();
298         final StreamResult result = new StreamResult(sw);
299         final DOMSource source = new DOMSource(doc);
300         trans.transform(source, result);
301         return sw.toString();
302     }

304     /**
305      * Print system information.
306      *
307      * @param args
308      *            variable arguments to pass to main. Valid values are "-show" and "-lookup".
309      */
310     public static void main(final String[] args) {
311         boolean show = false;
312         final List<String> lookup = new ArrayList<>();
313         if (args != null) {
314             String arg;
315             for (int i = 0; i < args.length; i++) {
316                 arg = args[i];
317                 if (null != arg) {
318                     switch (arg) {
319                         case "-show":
320                             show = true;
321                             break;
322                         case "-lookup":
323                             lookup.add(args[++i]);
324                             break;
325                         default:
326                             WaffleInfo.LOGGER.error("Unknown Argument: {}", arg);
327                             throw new RuntimeException("Unknown Argument: " + arg);
328                     }
329                 }
330             }
331         }
332 
333         final WaffleInfo">WaffleInfo helper = new WaffleInfo();
334         try {
335             final Document info = helper.getWaffleInfo();
336             for (final String name : lookup) {
337                 info.getDocumentElement().appendChild(helper.getLookupInfo(info, name));
338             }
339 
340             final String xml = WaffleInfo.toPrettyXML(info);
341             final File f;
342             if (show) {
343                 f = File.createTempFile("waffle-info-", ".xml");
344                 Files.write(f.toPath(), xml.getBytes(StandardCharsets.UTF_8), StandardOpenOption.APPEND);
345                 Desktop.getDesktop().open(f);
346             } else {
347                 WaffleInfo.LOGGER.info(xml);
348             }
349         } catch (final IOException | TransformerException | ParserConfigurationException e) {
350             WaffleInfo.LOGGER.error(e.getMessage());
351             WaffleInfo.LOGGER.trace("", e);
352         }
353     }
