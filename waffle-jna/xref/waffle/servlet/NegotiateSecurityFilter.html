97      /*
98       * (non-Javadoc)
99       * @see javax.servlet.Filter#doFilter(javax.servlet.ServletRequest, javax.servlet.ServletResponse,
100      * javax.servlet.FilterChain)
101      */
102     @Override
103     public void doFilter(final ServletRequest sreq, final ServletResponse sres, final FilterChain chain)
104             throws IOException, ServletException {
105 
106         final HttpServletRequest request = (HttpServletRequest) sreq;
107         final HttpServletResponse response = (HttpServletResponse) sres;
108 
109         NegotiateSecurityFilter.LOGGER.debug("{} {}, contentlength: {}", request.getMethod(), request.getRequestURI(),
110                 Integer.valueOf(request.getContentLength()));
111 
112         if (this.doFilterPrincipal(request, response, chain)) {
113             // previously authenticated user
114             return;
115         }
116 
117         final AuthorizationHeader authorizationHeader = new AuthorizationHeader(request);
118 
119         // authenticate user
120         if (!authorizationHeader.isNull()) {
121 
122             // log the user in using the token
123             IWindowsIdentity windowsIdentity;
124             try {
125                 windowsIdentity = this.providers.doFilter(request, response);
126                 if (windowsIdentity == null) {
127                     return;
128                 }
129             } catch (final IOException e) {
130                 NegotiateSecurityFilter.LOGGER.warn("error logging in user: {}", e.getMessage());
131                 NegotiateSecurityFilter.LOGGER.trace("{}", e);
132                 this.sendUnauthorized(response, true);
133                 return;
134             }
135 
136             IWindowsImpersonationContext ctx = null;
137             try {
138                 if (!this.allowGuestLogin && windowsIdentity.isGuest()) {
139                     NegotiateSecurityFilter.LOGGER.warn("guest login disabled: {}", windowsIdentity.getFqn());
140                     this.sendUnauthorized(response, true);
141                     return;
142                 }
143 
144                 NegotiateSecurityFilter.LOGGER.debug("logged in user: {} ({})", windowsIdentity.getFqn(),
145                         windowsIdentity.getSidString());
146 
147                 final HttpSession session = request.getSession(true);
148                 if (session == null) {
149                     throw new ServletException("Expected HttpSession");
150                 }
151 
152                 Subject subject = (Subject) session.getAttribute("javax.security.auth.subject");
153                 if (subject == null) {
154                     subject = new Subject();
155                 }
156 
157                 WindowsPrincipal windowsPrincipal;
158                 if (this.impersonate) {
159                     windowsPrincipal = new AutoDisposableWindowsPrincipal(windowsIdentity, this.principalFormat,
160                             this.roleFormat);
161                 } else {
162                     windowsPrincipal = new WindowsPrincipal(windowsIdentity, this.principalFormat, this.roleFormat);
163                 }
164 
165                 NegotiateSecurityFilter.LOGGER.debug("roles: {}", windowsPrincipal.getRolesString());
166                 subject.getPrincipals().add(windowsPrincipal);
167                 session.setAttribute("javax.security.auth.subject", subject);
168 
169                 NegotiateSecurityFilter.LOGGER.info("successfully logged in user: {}", windowsIdentity.getFqn());
170 
171                 request.getSession().setAttribute(NegotiateSecurityFilter.PRINCIPALSESSIONKEY, windowsPrincipal);
172 
173                 final NegotiateRequestWrapper requestWrapper = new NegotiateRequestWrapper(request, windowsPrincipal);
174 
175                 if (this.impersonate) {
176                     NegotiateSecurityFilter.LOGGER.debug("impersonating user");
177                     ctx = windowsIdentity.impersonate();
178                 }
179 
180                 chain.doFilter(requestWrapper, response);
181             } finally {
182                 if (this.impersonate && ctx != null) {
183                     NegotiateSecurityFilter.LOGGER.debug("terminating impersonation");
184                     ctx.revertToSelf();
185                 } else {
186                     windowsIdentity.dispose();
187                 }
188             }
189 
190             return;
191         }
192 
193         NegotiateSecurityFilter.LOGGER.debug("authorization required");
194         this.sendUnauthorized(response, false);
195     }

197     /**
198      * Filter for a previously logged on user.
199      *
200      * @param request
201      *            HTTP request.
202      * @param response
203      *            HTTP response.
204      * @param chain
205      *            Filter chain.
206      * @return True if a user already authenticated.
207      * @throws IOException
208      *             Signals that an I/O exception has occurred.
209      * @throws ServletException
210      *             the servlet exception
211      */
212     private boolean doFilterPrincipal(final HttpServletRequest request, final HttpServletResponse response,
213             final FilterChain chain) throws IOException, ServletException {
214         Principal principal = request.getUserPrincipal();
215         if (principal == null) {
216             final HttpSession session = request.getSession(false);
217             if (session != null) {
218                 principal = (Principal) session.getAttribute(NegotiateSecurityFilter.PRINCIPALSESSIONKEY);
219             }
220         }
221 
222         if (principal == null) {
223             // no principal in this request
224             return false;
225         }
226 
227         if (this.providers.isPrincipalException(request)) {
228             // the providers signal to authenticate despite an existing principal, eg. NTLM post
229             return false;
230         }
231 
232         // user already authenticated
233 
234         if (principal instanceof WindowsPrincipal) {
235             NegotiateSecurityFilter.LOGGER.debug("previously authenticated Windows user: {}", principal.getName());
236             final WindowsPrincipal windowsPrincipal = (WindowsPrincipal) principal;
237 
238             if (this.impersonate && windowsPrincipal.getIdentity() == null) {
239                 // This can happen when the session has been serialized then de-serialized
240                 // and because the IWindowsIdentity field is transient. In this case re-ask an
241                 // authentication to get a new identity.
242                 return false;
243             }
244 
245             final NegotiateRequestWrapper requestWrapper = new NegotiateRequestWrapper(request, windowsPrincipal);
246 
247             IWindowsImpersonationContext ctx = null;
248             if (this.impersonate) {
249                 NegotiateSecurityFilter.LOGGER.debug("re-impersonating user");
250                 ctx = windowsPrincipal.getIdentity().impersonate();
251             }
252             try {
253                 chain.doFilter(requestWrapper, response);
254             } finally {
255                 if (this.impersonate && ctx != null) {
256                     NegotiateSecurityFilter.LOGGER.debug("terminating impersonation");
257                     ctx.revertToSelf();
258                 }
259             }
260         } else {
261             NegotiateSecurityFilter.LOGGER.debug("previously authenticated user: {}", principal.getName());
262             chain.doFilter(request, response);
263         }
264         return true;
265     }

267     /*
268      * (non-Javadoc)
269      * @see javax.servlet.Filter#init(javax.servlet.FilterConfig)
270      */
271     @SuppressWarnings("unchecked")
272     @Override
273     public void init(final FilterConfig filterConfig) throws ServletException {
274         final Map<String, String> implParameters = new HashMap<>();
275 
276         String authProvider = null;
277         String[] providerNames = null;
278         if (filterConfig != null) {
279             final Enumeration<String> parameterNames = filterConfig.getInitParameterNames();
280             while (parameterNames.hasMoreElements()) {
281                 final String parameterName = parameterNames.nextElement();
282                 final String parameterValue = filterConfig.getInitParameter(parameterName);
283                 NegotiateSecurityFilter.LOGGER.debug("{}={}", parameterName, parameterValue);
284                 switch (parameterName) {
285                     case "principalFormat":
286                         this.principalFormat = PrincipalFormat.valueOf(parameterValue.toUpperCase(Locale.ENGLISH));
287                         break;
288                     case "roleFormat":
289                         this.roleFormat = PrincipalFormat.valueOf(parameterValue.toUpperCase(Locale.ENGLISH));
290                         break;
291                     case "allowGuestLogin":
292                         this.allowGuestLogin = Boolean.parseBoolean(parameterValue);
293                         break;
294                     case "impersonate":
295                         this.impersonate = Boolean.parseBoolean(parameterValue);
296                         break;
297                     case "securityFilterProviders":
298                         providerNames = parameterValue.split("\\s+");
299                         break;
300                     case "authProvider":
301                         authProvider = parameterValue;
302                         break;
303                     default:
304                         implParameters.put(parameterName, parameterValue);
305                         break;
306                 }
307             }
308         }
309 
310         if (authProvider != null) {
311             try {
312                 this.auth = (IWindowsAuthProvider) Class.forName(authProvider).getConstructor().newInstance();
313             } catch (final ClassNotFoundException | IllegalArgumentException | SecurityException
314                     | InstantiationException | IllegalAccessException | InvocationTargetException
315                     | NoSuchMethodException e) {
316                 NegotiateSecurityFilter.LOGGER.error("error loading '{}': {}", authProvider, e.getMessage());
317                 NegotiateSecurityFilter.LOGGER.trace("{}", e);
318                 throw new ServletException(e);
319             }
320         }
321 
322         if (this.auth == null) {
323             this.auth = new WindowsAuthProviderImpl();
324         }
325 
326         if (providerNames != null) {
327             this.providers = new SecurityFilterProviderCollection(providerNames, this.auth);
328         }
329 
330         // create default providers if none specified
331         if (this.providers == null) {
332             NegotiateSecurityFilter.LOGGER.debug("initializing default security filter providers");
333             this.providers = new SecurityFilterProviderCollection(this.auth);
334         }
335 
336         // apply provider implementation parameters
337         for (final Entry<String, String> implParameter : implParameters.entrySet()) {
338             final String[] classAndParameter = implParameter.getKey().split("/", 2);
339             if (classAndParameter.length == 2) {
340                 try {
341 
342                     NegotiateSecurityFilter.LOGGER.debug("setting {}, {}={}", classAndParameter[0],
343                             classAndParameter[1], implParameter.getValue());
344 
345                     final SecurityFilterProvider provider = this.providers.getByClassName(classAndParameter[0]);
346                     provider.initParameter(classAndParameter[1], implParameter.getValue());
347 
348                 } catch (final ClassNotFoundException e) {
349                     NegotiateSecurityFilter.LOGGER.error("invalid class: {} in {}", classAndParameter[0],
350                             implParameter.getKey());
351                     throw new ServletException(e);
352                 } catch (final Exception e) {
353                     NegotiateSecurityFilter.LOGGER.error("{}: error setting '{}': {}", classAndParameter[0],
354                             classAndParameter[1], e.getMessage());
355                     NegotiateSecurityFilter.LOGGER.trace("{}", e);
356                     throw new ServletException(e);
357                 }
358             } else {
359                 NegotiateSecurityFilter.LOGGER.error("Invalid parameter: {}", implParameter.getKey());
360                 throw new ServletException("Invalid parameter: " + implParameter.getKey());
361             }
362         }
363 
364         NegotiateSecurityFilter.LOGGER.info("[waffle.servlet.NegotiateSecurityFilter] started");
365     }

407     /**
408      * Send a 401 Unauthorized along with protocol authentication headers.
409      * 
410      * @param response
411      *            HTTP Response
412      * @param close
413      *            Close connection.
414      */
415     private void sendUnauthorized(final HttpServletResponse response, final boolean close) {
416         try {
417             this.providers.sendUnauthorized(response);
418             if (close) {
419                 response.setHeader("Connection", "close");
420             } else {
421                 response.setHeader("Connection", "keep-alive");
422             }
423             response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
424             response.flushBuffer();
425         } catch (final IOException e) {
426             throw new RuntimeException(e);
427         }
428     }
