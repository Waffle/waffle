24  package waffle.windows.auth;
25  
26  /**
27   * Implements Windows authentication functions.
28   *
29   * @author dblock[at]dblock[dot]org
30   */
31  public interface IWindowsAuthProvider {
32  
33      /**
34       * The LogonUser function attempts to log a user on to the local computer using a network logon type and the default
35       * authentication provider.
36       *
37       * @param username
38       *            A string that specifies the name of the user in the UPN format.
39       * @param password
40       *            A string that specifies the plaintext password for the user account specified by username.
41       * @return Windows identity.
42       */
43      IWindowsIdentity logonUser(final String username, final String password);
44  
45      /**
46       * The LogonDomainUser function attempts to log a user on to the local computer using a network logon type and the
47       * default authentication provider.
48       *
49       * @param username
50       *            A string that specifies the name of the user. This is the name of the user account to log on to. If
51       *            you use the user principal name (UPN) format, user@DNS_domain_name, the domain parameter must be NULL.
52       * @param domain
53       *            A string that specifies the name of the domain or server whose account database contains the username
54       *            account. If this parameter is NULL, the user name must be specified in UPN format. If this parameter
55       *            is ".", the function validates the account by using only the local account database.
56       * @param password
57       *            A string that specifies the plaintext password for the user account specified by username.
58       * @return Windows identity.
59       */
60      IWindowsIdentity logonDomainUser(final String username, final String domain, final String password);
61  
62      /**
63       * The LogonDomainUserEx function attempts to log a user on to the local computer. The local computer is the
64       * computer from which LogonUser was called. You cannot use LogonUser to log on to a remote computer. You specify
65       * the user with a user name and domain and authenticate the user with a plaintext password.
66       *
67       * @param username
68       *            A string that specifies the name of the user. This is the name of the user account to log on to. If
69       *            you use the user principal name (UPN) format, user@DNS_domain_name, the domain parameter must be NULL.
70       * @param domain
71       *            A string that specifies the name of the domain or server whose account database contains the username
72       *            account. If this parameter is NULL, the user name must be specified in UPN format. If this parameter
73       *            is ".", the function validates the account by using only the local account database.
74       * @param password
75       *            A string that specifies the plaintext password for the user account specified by username.
76       * @param logonType
77       *            The type of logon operation to perform.
78       * @param logonProvider
79       *            Specifies the logon provider.
80       * @return Windows identity.
81       */
82      IWindowsIdentity logonDomainUserEx(final String username, final String domain, final String password,
83              final int logonType, final int logonProvider);
84  
85      /**
86       * Retrieve a security identifier (SID) for the account and the name of the domain or local computer on which the
87       * account was found.
88       *
89       * @param username
90       *            Fully qualified or partial username.
91       * @return Windows account.
92       */
93      IWindowsAccount lookupAccount(final String username);
94  
95      /**
96       * Retrieve the current computer information.
97       *
98       * @return Current computer information.
99       */
100     IWindowsComputer getCurrentComputer();
101 
102     /**
103      * Retrieve a list of domains (Active Directory) on the local server.
104      *
105      * @return A list of domains.
106      */
107     IWindowsDomain[] getDomains();
108 
109     /**
110      * Attempts to validate the user using an SSPI token. This token is generated by the client via the
111      * InitializeSecurityContext(package) method described in
112      * https://msdn.microsoft.com/en-us/library/aa375509(VS.85).aspx
113      *
114      * @param connectionId
115      *            A unique connection id.
116      * @param token
117      *            The security token generated by the client wishing to logon.
118      * @param securityPackage
119      *            The name of the security package to use. Can be any security package supported by both the client and
120      *            the server. This is usually set to "Negotiate" which will use SPNEGO to determine which security
121      *            package to use. Other common values are "Kerberos" and "NTLM".
122      * @return Windows account.
123      */
124     IWindowsSecurityContext acceptSecurityToken(final String connectionId, final byte[] token,
125             final String securityPackage);
126 
127     /**
128      * Reset a previously saved continuation security token for a given connection id.
129      *
130      * @param connectionId
131      *            Connection id.
132      */
133     void resetSecurityToken(final String connectionId);
134 }
