24  package waffle.windows.auth.impl;
25  
26  import com.sun.jna.platform.win32.Secur32;
27  import com.sun.jna.platform.win32.Sspi.CtxtHandle;
28  import com.sun.jna.platform.win32.Win32Exception;
29  import com.sun.jna.platform.win32.WinError;
30  
31  import waffle.windows.auth.IWindowsImpersonationContext;
32  
33  /**
34   * The Class WindowsSecurityContextImpersonationContextImpl.
35   */
36  public class WindowsSecurityContextImpersonationContextImpl implements IWindowsImpersonationContext {
37  
38      /** The ctx. */
39      private final CtxtHandle ctx;
40  
41      /**
42       * Instantiates a new windows security context impersonation context impl.
43       *
44       * @param newCtx
45       *            the new ctx
46       */
47      public WindowsSecurityContextImpersonationContextImpl(final CtxtHandle newCtx) {
48          final int rc = Secur32.INSTANCE.ImpersonateSecurityContext(newCtx);
49          if (rc != WinError.SEC_E_OK) {
50              throw new Win32Exception(rc);
51          }
52  
53          this.ctx = newCtx;
54      }
55  
56      @Override
57      public void revertToSelf() {
58          final int rc = Secur32.INSTANCE.RevertSecurityContext(this.ctx);
59          if (rc != WinError.SEC_E_OK) {
60              throw new Win32Exception(rc);
61          }
62      }
63  }
