61      @Override
62      public IWindowsIdentity getIdentity() {
63          final HANDLEByReference phContextToken = new HANDLEByReference();
64          final int rc = Secur32.INSTANCE.QuerySecurityContextToken(this.ctx, phContextToken);
65          if (WinError.SEC_E_OK != rc) {
66              throw new Win32Exception(rc);
67          }
68          return new WindowsIdentityImpl(phContextToken.getValue());
69      }

76      @Override
77      public byte[] getToken() {
78          return this.token == null || this.token.getBytes() == null ? null : this.token.getBytes().clone();
79      }

81      /**
82       * Get the current Windows security context for a given SSPI package.
83       * 
84       * @param securityPackage
85       *            SSPI package.
86       * @param targetName
87       *            The target of the context. The string contents are security-package specific.
88       * @return Windows security context.
89       */
90      public static IWindowsSecurityContext getCurrent(final String securityPackage, final String targetName) {
91          IWindowsCredentialsHandle credentialsHandle = WindowsCredentialsHandleImpl.getCurrent(securityPackage);
92          credentialsHandle.initialize();
93          try {
94              final WindowsSecurityContextImpl ctx = new WindowsSecurityContextImpl();
95              ctx.setPrincipalName(WindowsAccountImpl.getCurrentUsername());
96              ctx.setCredentialsHandle(credentialsHandle);
97              ctx.setSecurityPackage(securityPackage);
98              ctx.initialize(null, null, targetName);
99  
100             // Starting from here ctx 'owns' the credentials handle, so let's null out the
101             // variable. This will prevent the finally block below from disposing it right away.
102             credentialsHandle = null;
103 
104             return ctx;
105         } finally {
106             if (credentialsHandle != null) {
107                 credentialsHandle.dispose();
108             }
109         }
110     }
111 
112     @Override
113     public void initialize(final CtxtHandle continueCtx, final SecBufferDesc continueToken, final String targetName) {
114         this.attr = new IntByReference();
115         this.ctx = new CtxtHandle();
116         int tokenSize = Sspi.MAX_TOKEN_SIZE;
117         int rc;
118         do {
119             this.token = new SecBufferDesc(Sspi.SECBUFFER_TOKEN, tokenSize);
120             rc = Secur32.INSTANCE.InitializeSecurityContext(this.credentials.getHandle(), continueCtx, targetName,
121                     Sspi.ISC_REQ_CONNECTION, 0, Sspi.SECURITY_NATIVE_DREP, continueToken, 0, this.ctx, this.token,
122                     this.attr, null);
123             switch (rc) {
124                 case WinError.SEC_E_INSUFFICIENT_MEMORY:
125                 case WinError.SEC_E_BUFFER_TOO_SMALL:
126                     tokenSize += Sspi.MAX_TOKEN_SIZE;
127                     break;
128                 case WinError.SEC_I_CONTINUE_NEEDED:
129                     this.continueFlag = true;
130                     break;
131                 case WinError.SEC_E_OK:
132                     this.continueFlag = false;
133                     break;
134                 default:
135                     throw new Win32Exception(rc);
136             }
137         } while (rc == WinError.SEC_E_INSUFFICIENT_MEMORY || rc == WinError.SEC_E_BUFFER_TOO_SMALL);
138     }
139 
140     @Override
141     public void dispose() {
142         WindowsSecurityContextImpl.dispose(this.ctx);
143 
144         if (this.credentials != null) {
145             this.credentials.dispose();
146         }
147     }
148 
149     /**
150      * Dispose a security context.
151      * 
152      * @param ctx
153      *            Security context.
154      * @return True if a context was disposed.
155      */
156     public static boolean dispose(final CtxtHandle ctx) {
157         if (ctx != null && !ctx.isNull()) {
158             final int rc = Secur32.INSTANCE.DeleteSecurityContext(ctx);
159             if (WinError.SEC_E_OK != rc) {
160                 throw new Win32Exception(rc);
161             }
162             return true;
163         }
164         return false;
165     }
