24  package waffle.spring.boot.demo;
25  
26  import org.springframework.security.config.annotation.web.builders.HttpSecurity;
27  import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
28  import org.springframework.security.config.annotation.web.configuration.WebSecurityConfigurerAdapter;
29  import org.springframework.security.web.authentication.www.BasicAuthenticationFilter;
30  
31  import waffle.spring.NegotiateSecurityFilter;
32  import waffle.spring.NegotiateSecurityFilterEntryPoint;
33  
34  /**
35   * Demo Spring Boot Security configuration that configures the Negotiate filter to require authentication for all
36   * requests.
37   */
38  @EnableWebSecurity
39  public class SecurityConfig extends WebSecurityConfigurerAdapter {
40  
41      private NegotiateSecurityFilter filter;
42      private NegotiateSecurityFilterEntryPoint entryPoint;
43  
44      /**
45       * Autowire constructor injects bean auto-configured by Starter.
46       *
47       * @param filter
48       *            the filter
49       * @param entryPoint
50       *            the entry point
51       */
52      public SecurityConfig(NegotiateSecurityFilter filter, NegotiateSecurityFilterEntryPoint entryPoint) {
53          this.filter = filter;
54          this.entryPoint = entryPoint;
55      }
56  
57      @Override
58      protected void configure(HttpSecurity http) throws Exception {
59          http.authorizeRequests().anyRequest().authenticated().and()
60                  .addFilterBefore(filter, BasicAuthenticationFilter.class).exceptionHandling()
61                  .authenticationEntryPoint(entryPoint);
62      }
63  
64  }
