12  package waffle.spring.boot.demo;
13  
14  import java.util.stream.Collectors;
15  
16  import org.springframework.security.core.Authentication;
17  import org.springframework.web.bind.annotation.RequestMapping;
18  import org.springframework.web.bind.annotation.RestController;
19  
20  @RestController
21  public class DemoController {
22  
23      @RequestMapping
24      public String demo(Authentication auth) {
25          return String.format("Hello, %s. You have authorities: %s", auth.getPrincipal(),
26                  auth.getAuthorities().stream().map(a -> a.getAuthority()).collect(Collectors.joining(", ")));
27      }
28  
29  }
