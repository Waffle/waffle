14  package waffle.spring;
15  
16  import org.springframework.security.core.GrantedAuthority;
17  import org.springframework.security.core.authority.SimpleGrantedAuthority;
18  
19  import waffle.windows.auth.WindowsAccount;
20  
21  /**
22   * A {@link GrantedAuthorityFactory} that uses the {@link WindowsAccount}'s fqn as the basis of the
23   * {@link GrantedAuthority} string, and (optionally) applies two transformations:
24   * <ul>
25   * <li>prepending a prefix, and</li>
26   * <li>converting to uppercase</li>
27   * </ul>
28   */
29  public class FqnGrantedAuthorityFactory implements GrantedAuthorityFactory {
30  
31      private final String  prefix;
32      private final boolean convertToUpperCase;
33  
34      public FqnGrantedAuthorityFactory(final String newPrefix, final boolean newConvertToUpperCase) {
35          this.prefix = newPrefix;
36          this.convertToUpperCase = newConvertToUpperCase;
37      }
38  
39      @Override
40      public GrantedAuthority createGrantedAuthority(final WindowsAccount windowsAccount) {
41  
42          String grantedAuthorityString = windowsAccount.getFqn();
43  
44          if (this.prefix != null) {
45              grantedAuthorityString = this.prefix + grantedAuthorityString;
46          }
47  
48          if (this.convertToUpperCase) {
49              grantedAuthorityString = grantedAuthorityString.toUpperCase();
50          }
51  
52          return new SimpleGrantedAuthority(grantedAuthorityString);
53      }
54  }
