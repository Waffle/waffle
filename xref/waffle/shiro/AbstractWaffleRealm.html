37  /**
38   * A {@link org.apache.shiro.realm.Realm} that authenticates with Active Directory using WAFFLE. Authorization is left
39   * for subclasses to define by implementing the {@link #buildAuthorizationInfo} method.
40   */
41  public abstract class AbstractWaffleRealm extends AuthorizingRealm {
42      private static final Logger  LOGGER     = LoggerFactory.getLogger(AbstractWaffleRealm.class);
43      private static final String  REALM_NAME = "WAFFLE";
44  
45      private IWindowsAuthProvider provider   = new WindowsAuthProviderImpl();
46  
47      @Override
48      protected final AuthenticationInfo doGetAuthenticationInfo(final AuthenticationToken authToken) {
49          AuthenticationInfo authenticationInfo = null;
50          if (authToken instanceof UsernamePasswordToken) {
51              final UsernamePasswordToken token = (UsernamePasswordToken) authToken;
52              final String username = token.getUsername();
53              IWindowsIdentity identity = null;
54              try {
55                  LOGGER.debug("Attempting login for user {}", username);
56                  identity = this.provider.logonUser(username, new String(token.getPassword()));
57                  if (identity.isGuest()) {
58                      LOGGER.debug("Guest identity for user {}; denying access", username);
59                      throw new AuthenticationException("Guest identities are not allowed access");
60                  }
61                  final Object principal = new WaffleFqnPrincipal(identity);
62                  authenticationInfo = buildAuthenticationInfo(token, principal);
63                  LOGGER.debug("Successful login for user {}", username);
64              } catch (RuntimeException e) {
65                  LOGGER.debug("Failed login for user {}: {}", username, e.getMessage());
66                  LOGGER.trace("{}", e);
67                  throw new AuthenticationException("Login failed", e);
68              } finally {
69                  if (identity != null) {
70                      identity.dispose();
71                  }
72              }
73          }
74          return authenticationInfo;
75      }
76  
77      private AuthenticationInfo buildAuthenticationInfo(final UsernamePasswordToken token, final Object principal) {
78          AuthenticationInfo authenticationInfo;
79          final HashingPasswordService hashService = getHashService();
80          if (hashService != null) {
81              final Hash hash = hashService.hashPassword(token.getPassword());
82              final ByteSource salt = hash.getSalt();
83              authenticationInfo = new SimpleAuthenticationInfo(principal, hash, salt, REALM_NAME);
84          } else {
85              final Object creds = token.getCredentials();
86              authenticationInfo = new SimpleAuthenticationInfo(principal, creds, REALM_NAME);
87          }
88          return authenticationInfo;
89      }
90  
91      @Override
92      protected final AuthorizationInfo doGetAuthorizationInfo(final PrincipalCollection principals) {
93          final WaffleFqnPrincipal principal = principals.oneByType(WaffleFqnPrincipal.class);
94          return principal == null ? null : buildAuthorizationInfo(principal);
95      }
96  
97      /**
98       * Assembles the appropriate authorization information for the specified principal.
99       * 
100      * @param principal
101      *            the principal for which to assemble authorization information
102      * @return the authorization information for the specified principal
103      */
104     protected abstract AuthorizationInfo buildAuthorizationInfo(final WaffleFqnPrincipal principal);
105 
106     /**
107      * Allow overriding the default implementation of {@link IWindowsAuthProvider} This is only needed for testing,
108      * since for normal usage the default is what you want.
109      * 
110      * @param value
111      *            the windows authorization provider
112      */
113     void setProvider(final IWindowsAuthProvider value) {
114         this.provider = value;
115     }
116 
117     private HashingPasswordService getHashService() {
118         final CredentialsMatcher matcher = getCredentialsMatcher();
119         if (matcher instanceof PasswordMatcher) {
120             final PasswordMatcher passwordMatcher = (PasswordMatcher) matcher;
121             final PasswordService passwordService = passwordMatcher.getPasswordService();
122             if (passwordService instanceof HashingPasswordService) {
123                 return (HashingPasswordService) passwordService;
124             }
125         }
126         return null;
127     }
