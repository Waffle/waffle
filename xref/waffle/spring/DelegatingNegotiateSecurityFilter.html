31  /**
32   *
33   * <p>
34   * Supports optional injection of spring security entities, allowing Waffle to act as an interface towards an identity
35   * provider(the AD).
36   * </p>
37   *
38   * <i>Below mentioned entities are verified to be set before invoked, inherited entities are not.</i>
39   *
40   * <ul>
41   * <li>The <code>AuthenticationManager</code> allows for the service provider to authorize the principal.</li>
42   *
43   * <li>The <code>authenticationSuccessHandler</code> allows for the service provider to further populate the
44   * {@link org.springframework.security.core.Authentication Authentication} object.</li>
45   *
46   * <li>The <code>AuthenticationFailureHandler</code> is called if the AuthenticationManager throws an
47   * {@link org.springframework.security.core.AuthenticationException AuthenticationException}.</li>
48   *
49   * <li>The <code>AccessDeniedHandler</code> is called if the AuthenticationManager throws an
50   * {@link org.springframework.security.access.AccessDeniedException AccessDeniedException}.</li>
51   * </ul>
52   * Example configuration:
53   *
54   * <pre>
55   * {@code
56   * <bean id="waffleNegotiateSecurityFilter"
57   *      class="waffle.spring.DelegatingNegotiateSecurityFilter"
58   *      scope="tenant">
59   *      <property name="allowGuestLogin" value="false" />
60   *      <property name="Provider" ref="waffleSecurityFilterProviderCollection" />
61   *      <property name="authenticationManager" ref="authenticationManager" />
62   *      <property name="authenticationSuccessHandler" ref="authenticationSuccessHandler" />
63   *      <property name="authenticationFailureHandler" ref="authenticationFailureHandler" />
64   *      <property name="accessDeniedHandler" ref="accessDeniedHandler" />
65   *      <property name="defaultGrantedAuthority">
66   *          <null />
67   *      </property>
68   * </bean>
69   * }
70   * </pre>
71   */
72  public class DelegatingNegotiateSecurityFilter extends NegotiateSecurityFilter {
73  
74      /** The Constant LOGGER. */
75      private static final Logger LOGGER = LoggerFactory.getLogger(DelegatingNegotiateSecurityFilter.class);

135     @Override
136     protected boolean setAuthentication(final HttpServletRequest request, final HttpServletResponse response,
137             final Authentication authentication) {
138         try {
139             if (this.authenticationManager != null) {
140                 DelegatingNegotiateSecurityFilter.LOGGER.debug("Delegating to custom authenticationmanager");
141                 final Authentication customAuthentication = this.authenticationManager.authenticate(authentication);
142                 SecurityContextHolder.getContext().setAuthentication(customAuthentication);
143             } else {
144                 SecurityContextHolder.getContext().setAuthentication(authentication);
145             }
146             if (this.authenticationSuccessHandler != null) {
147                 try {
148                     this.authenticationSuccessHandler.onAuthenticationSuccess(request, response, authentication);
149                 } catch (final IOException | ServletException e) {
150                     DelegatingNegotiateSecurityFilter.LOGGER.warn("Error calling authenticationSuccessHandler: {}",
151                             e.getMessage());
152                     DelegatingNegotiateSecurityFilter.LOGGER.trace("", e);
153                     return false;
154                 }
155             }
156         } catch (final AuthenticationException e) {
157             DelegatingNegotiateSecurityFilter.LOGGER
158                     .warn("Error authenticating user in custom authenticationmanager: {}", e.getMessage());
159             this.sendAuthenticationFailed(request, response, e);
160             return false;
161         } catch (final AccessDeniedException e) {
162             DelegatingNegotiateSecurityFilter.LOGGER.warn("Error authorizing user in custom authenticationmanager: {}",
163                     e.getMessage());
164             this.sendAccessDenied(request, response, e);
165             return false;
166         }
167         return true;
168     }
169 
170     @Override
171     public void afterPropertiesSet() throws ServletException {
172         super.afterPropertiesSet();
173 
174         if (this.getProvider() == null) {
175             throw new ServletException("Missing NegotiateSecurityFilter.Provider");
176         }
177     }
178 
179     /**
180      * Forward to authenticationFailureHandler.
181      *
182      * @param request
183      *            the request
184      * @param response
185      *            HTTP Response
186      * @param ae
187      *            the ae
188      */
189     private void sendAuthenticationFailed(final HttpServletRequest request, final HttpServletResponse response,
190             final AuthenticationException ae) {
191         if (this.authenticationFailureHandler != null) {
192             try {
193                 this.authenticationFailureHandler.onAuthenticationFailure(request, response, ae);
194                 return;
195             } catch (final IOException e) {
196                 DelegatingNegotiateSecurityFilter.LOGGER.warn("IOException invoking authenticationFailureHandler: {}",
197                         e.getMessage());
198                 DelegatingNegotiateSecurityFilter.LOGGER.trace("", e);
199             } catch (final ServletException e) {
200                 DelegatingNegotiateSecurityFilter.LOGGER
201                         .warn("ServletException invoking authenticationFailureHandler: {}", e.getMessage());
202                 DelegatingNegotiateSecurityFilter.LOGGER.trace("", e);
203             }
204         }
205         super.sendUnauthorized(response, true);
206     }
207 
208     /**
209      * Forward to accessDeniedHandler.
210      *
211      * @param request
212      *            the request
213      * @param response
214      *            HTTP Response
215      * @param ae
216      *            the ae
217      */
218     private void sendAccessDenied(final HttpServletRequest request, final HttpServletResponse response,
219             final AccessDeniedException ae) {
220         if (this.accessDeniedHandler != null) {
221             try {
222                 this.accessDeniedHandler.handle(request, response, ae);
223                 return;
224             } catch (final IOException e) {
225                 DelegatingNegotiateSecurityFilter.LOGGER.warn("IOException invoking accessDeniedHandler: {}",
226                         e.getMessage());
227                 DelegatingNegotiateSecurityFilter.LOGGER.trace("", e);
228             } catch (final ServletException e) {
229                 DelegatingNegotiateSecurityFilter.LOGGER.warn("ServletException invoking accessDeniedHandler: {}",
230                         e.getMessage());
231                 DelegatingNegotiateSecurityFilter.LOGGER.trace("", e);
232             }
233         }
234         // fallback
235         this.sendUnauthorized(response, true);
236     }
