105     @Override
106     public void doFilter(final ServletRequest sreq, final ServletResponse sres, final FilterChain chain)
107             throws IOException, ServletException {
108 
109         final HttpServletRequest request = (HttpServletRequest) sreq;
110         final HttpServletResponse response = (HttpServletResponse) sres;
111 
112         NegotiateSecurityFilter.LOGGER.debug("{} {}, contentlength: {}", request.getMethod(), request.getRequestURI(),
113                 Integer.valueOf(request.getContentLength()));
114 
115         // If we are not in a windows environment, resume filter chain
116         if (!NegotiateSecurityFilter.isWindows()) {
117             NegotiateSecurityFilter.LOGGER.debug("Running in a non windows environment, SSO skipped");
118             chain.doFilter(request, response);
119             return;
120         }
121 
122         // If sso is disabled, resume filter chain
123         if (this.disableSSO) {
124             NegotiateSecurityFilter.LOGGER.debug("SSO is disabled, resuming filter chain");
125             chain.doFilter(request, response);
126             return;
127         }
128 
129         // If excluded URL, resume the filter chain
130         if (request.getRequestURL() != null && this.excludePatterns != null) {
131             final String url = request.getRequestURL().toString();
132             for (final String pattern : this.excludePatterns) {
133                 if (url.matches(pattern)) {
134                     NegotiateSecurityFilter.LOGGER.info("Pattern :{} excluded URL:{}", url, pattern);
135                     chain.doFilter(sreq, sres);
136                     return;
137                 }
138             }
139         }
140 
141         // If exclude cores pre-flight and is pre flight, resume the filter chain
142         if (this.excludeCorsPreflight && CorsPreFlightCheck.isPreflight(request)) {
143             NegotiateSecurityFilter.LOGGER.debug("[waffle.servlet.NegotiateSecurityFilter] CORS preflight");
144             chain.doFilter(sreq, sres);
145             return;
146         }
147 
148         final AuthorizationHeaderthorizationHeader authorizationHeader = new AuthorizationHeader(request);
149 
150         // If exclude bearer authorization and is bearer authorization, result the filter chain
151         if (this.excludeBearerAuthorization && authorizationHeader.isBearerAuthorizationHeader()) {
152             NegotiateSecurityFilter.LOGGER.debug("[waffle.servlet.NegotiateSecurityFilter] Authorization: Bearer");
153             chain.doFilter(sreq, sres);
154             return;
155         }
156 
157         if (this.doFilterPrincipal(request, response, chain)) {
158             // previously authenticated user
159             return;
160         }
161 
162         // authenticate user
163         if (!authorizationHeader.isNull()) {
164 
165             // log the user in using the token
166             IWindowsIdentity windowsIdentity;
167             try {
168                 windowsIdentity = this.providers.doFilter(request, response);
169                 if (windowsIdentity == null) {
170                     return;
171                 }
172             } catch (final IOException e) {
173                 NegotiateSecurityFilter.LOGGER.warn("error logging in user: {}", e.getMessage());
174                 NegotiateSecurityFilter.LOGGER.trace("", e);
175                 this.sendUnauthorized(response, true);
176                 return;
177             }
178 
179             IWindowsImpersonationContext ctx = null;
180             try {
181                 if (!this.allowGuestLogin && windowsIdentity.isGuest()) {
182                     NegotiateSecurityFilter.LOGGER.warn("guest login disabled: {}", windowsIdentity.getFqn());
183                     this.sendUnauthorized(response, true);
184                     return;
185                 }
186 
187                 NegotiateSecurityFilter.LOGGER.debug("logged in user: {} ({})", windowsIdentity.getFqn(),
188                         windowsIdentity.getSidString());
189 
190                 final HttpSession session = request.getSession(true);
191                 if (session == null) {
192                     throw new ServletException("Expected HttpSession");
193                 }
194 
195                 Subject subject = (Subject) session.getAttribute("javax.security.auth.subject");
196                 if (subject == null) {
197                     subject = new Subject();
198                 }
199 
200                 WindowsPrincipal windowsPrincipal;
201                 if (this.impersonate) {
202                     windowsPrincipal = new AutoDisposableWindowsPrincipal(windowsIdentity, this.principalFormat,
203                             this.roleFormat);
204                 } else {
205                     windowsPrincipal = new WindowsPrincipal(windowsIdentity, this.principalFormat, this.roleFormat);
206                 }
207 
208                 NegotiateSecurityFilter.LOGGER.debug("roles: {}", windowsPrincipal.getRolesString());
209                 subject.getPrincipals().add(windowsPrincipal);
210                 request.getSession(false).setAttribute("javax.security.auth.subject", subject);
211 
212                 NegotiateSecurityFilter.LOGGER.info("successfully logged in user: {}", windowsIdentity.getFqn());
213 
214                 request.getSession(false).setAttribute(NegotiateSecurityFilter.PRINCIPALSESSIONKEY, windowsPrincipal);
215 
216                 final NegotiateRequestWrappertWrapper">NegotiateRequestWrapper requestWrapper = new NegotiateRequestWrapper(request, windowsPrincipal);
217 
218                 if (this.impersonate) {
219                     NegotiateSecurityFilter.LOGGER.debug("impersonating user");
220                     ctx = windowsIdentity.impersonate();
221                 }
222 
223                 chain.doFilter(requestWrapper, response);
224             } finally {
225                 if (this.impersonate && ctx != null) {
226                     NegotiateSecurityFilter.LOGGER.debug("terminating impersonation");
227                     ctx.revertToSelf();
228                 } else {
229                     windowsIdentity.dispose();
230                 }
231             }
232 
233             return;
234         }
235 
236         NegotiateSecurityFilter.LOGGER.debug("authorization required");
237         this.sendUnauthorized(response, false);
238     }

240     /**
241      * Filter for a previously logged on user.
242      *
243      * @param request
244      *            HTTP request.
245      * @param response
246      *            HTTP response.
247      * @param chain
248      *            Filter chain.
249      * @return True if a user already authenticated.
250      * @throws IOException
251      *             Signals that an I/O exception has occurred.
252      * @throws ServletException
253      *             the servlet exception
254      */
255     private boolean doFilterPrincipal(final HttpServletRequest request, final HttpServletResponse response,
256             final FilterChain chain) throws IOException, ServletException {
257         Principal principal = request.getUserPrincipal();
258         if (principal == null) {
259             final HttpSession session = request.getSession(false);
260             if (session != null) {
261                 principal = (Principal) session.getAttribute(NegotiateSecurityFilter.PRINCIPALSESSIONKEY);
262             }
263         }
264 
265         if (principal == null) {
266             // no principal in this request
267             return false;
268         }
269 
270         if (this.providers.isPrincipalException(request)) {
271             // the providers signal to authenticate despite an existing principal, eg. NTLM post
272             return false;
273         }
274 
275         // user already authenticated
276         if (principal instanceof WindowsPrincipal) {
277             NegotiateSecurityFilter.LOGGER.debug("previously authenticated Windows user: {}", principal.getName());
278             final WindowsPrincipal/WindowsPrincipal.html#WindowsPrincipal">WindowsPrincipal windowsPrincipal = (WindowsPrincipal) principal;
279 
280             if (this.impersonate && windowsPrincipal.getIdentity() == null) {
281                 // This can happen when the session has been serialized then de-serialized
282                 // and because the IWindowsIdentity field is transient. In this case re-ask an
283                 // authentication to get a new identity.
284                 return false;
285             }
286 
287             final NegotiateRequestWrappertWrapper">NegotiateRequestWrapper requestWrapper = new NegotiateRequestWrapper(request, windowsPrincipal);
288 
289             IWindowsImpersonationContext ctx = null;
290             if (this.impersonate) {
291                 NegotiateSecurityFilter.LOGGER.debug("re-impersonating user");
292                 ctx = windowsPrincipal.getIdentity().impersonate();
293             }
294             try {
295                 chain.doFilter(requestWrapper, response);
296             } finally {
297                 if (this.impersonate && ctx != null) {
298                     NegotiateSecurityFilter.LOGGER.debug("terminating impersonation");
299                     ctx.revertToSelf();
300                 }
301             }
302         } else {
303             NegotiateSecurityFilter.LOGGER.debug("previously authenticated user: {}", principal.getName());
304             chain.doFilter(request, response);
305         }
306         return true;
307     }

309     @Override
310     public void init(final FilterConfig filterConfig) throws ServletException {
311         final Map<String, String> implParameters = new HashMap<>();
312 
313         NegotiateSecurityFilter.LOGGER.debug("[waffle.servlet.NegotiateSecurityFilter] starting");
314 
315         String authProvider = null;
316         String[] providerNames = null;
317         if (filterConfig != null) {
318             final Enumeration<String> parameterNames = filterConfig.getInitParameterNames();
319             NegotiateSecurityFilter.LOGGER.debug("[waffle.servlet.NegotiateSecurityFilter] processing filterConfig");
320             while (parameterNames.hasMoreElements()) {
321                 final String parameterName = parameterNames.nextElement();
322                 final String parameterValue = filterConfig.getInitParameter(parameterName);
323                 NegotiateSecurityFilter.LOGGER.debug("Init Param: '{}={}'", parameterName, parameterValue);
324                 switch (parameterName) {
325                     case "principalFormat":
326                         this.principalFormat = PrincipalFormat.valueOf(parameterValue.toUpperCase(Locale.ENGLISH));
327                         break;
328                     case "roleFormat":
329                         this.roleFormat = PrincipalFormat.valueOf(parameterValue.toUpperCase(Locale.ENGLISH));
330                         break;
331                     case "allowGuestLogin":
332                         this.allowGuestLogin = Boolean.parseBoolean(parameterValue);
333                         break;
334                     case "impersonate":
335                         this.impersonate = Boolean.parseBoolean(parameterValue);
336                         break;
337                     case "securityFilterProviders":
338                         providerNames = parameterValue.split("\\s+");
339                         break;
340                     case "authProvider":
341                         authProvider = parameterValue;
342                         break;
343                     case "excludePatterns":
344                         this.excludePatterns = parameterValue.split("\\s+");
345                         break;
346                     case "excludeCorsPreflight":
347                         this.excludeCorsPreflight = Boolean.parseBoolean(parameterValue);
348                         break;
349                     case "excludeBearerAuthorization":
350                         this.excludeBearerAuthorization = Boolean.parseBoolean(parameterValue);
351                         break;
352                     case "disableSSO":
353                         this.disableSSO = Boolean.parseBoolean(parameterValue);
354                         break;
355                     default:
356                         implParameters.put(parameterName, parameterValue);
357                         break;
358                 }
359             }
360         }
361 
362         NegotiateSecurityFilter.LOGGER.debug("[waffle.servlet.NegotiateSecurityFilter] authProvider");
363         if (authProvider != null) {
364             try {
365                 this.auth = (IWindowsAuthProvider) Class.forName(authProvider).getConstructor().newInstance();
366             } catch (final ClassNotFoundException | IllegalArgumentException | SecurityException
367                     | InstantiationException | IllegalAccessException | InvocationTargetException
368                     | NoSuchMethodException e) {
369                 NegotiateSecurityFilter.LOGGER.error("error loading '{}': {}", authProvider, e.getMessage());
370                 NegotiateSecurityFilter.LOGGER.trace("", e);
371                 throw new ServletException(e);
372             }
373         }
374 
375         if (this.auth == null) {
376             this.auth = new WindowsAuthProviderImpl();
377         }
378 
379         if (providerNames != null) {
380             this.providers = new SecurityFilterProviderCollection(providerNames, this.auth);
381         }
382 
383         // create default providers if none specified
384         if (this.providers == null) {
385             NegotiateSecurityFilter.LOGGER.debug("initializing default security filter providers");
386             this.providers = new SecurityFilterProviderCollection(this.auth);
387         }
388 
389         // apply provider implementation parameters
390         NegotiateSecurityFilter.LOGGER.debug("[waffle.servlet.NegotiateSecurityFilter] load provider parameters");
391         for (final Entry<String, String> implParameter : implParameters.entrySet()) {
392             final String[] classAndParameter = implParameter.getKey().split("/", 2);
393             if (classAndParameter.length == 2) {
394                 try {
395 
396                     NegotiateSecurityFilter.LOGGER.debug("setting {}, {}={}", classAndParameter[0],
397                             classAndParameter[1], implParameter.getValue());
398 
399                     final SecurityFilterProvider provider = this.providers.getByClassName(classAndParameter[0]);
400                     provider.initParameter(classAndParameter[1], implParameter.getValue());
401 
402                 } catch (final ClassNotFoundException e) {
403                     NegotiateSecurityFilter.LOGGER.error("invalid class: {} in {}", classAndParameter[0],
404                             implParameter.getKey());
405                     throw new ServletException(e);
406                 } catch (final Exception e) {
407                     NegotiateSecurityFilter.LOGGER.error("{}: error setting '{}': {}", classAndParameter[0],
408                             classAndParameter[1], e.getMessage());
409                     NegotiateSecurityFilter.LOGGER.trace("", e);
410                     throw new ServletException(e);
411                 }
412             } else {
413                 NegotiateSecurityFilter.LOGGER.error("Invalid parameter: {}", implParameter.getKey());
414                 throw new ServletException("Invalid parameter: " + implParameter.getKey());
415             }
416         }
417 
418         NegotiateSecurityFilter.LOGGER.info("[waffle.servlet.NegotiateSecurityFilter] started");
419     }

461     /**
462      * Send a 401 Unauthorized along with protocol authentication headers.
463      *
464      * @param response
465      *            HTTP Response
466      * @param close
467      *            Close connection.
468      */
469     private void sendUnauthorized(final HttpServletResponse response, final boolean close) {
470         try {
471             this.providers.sendUnauthorized(response);
472             if (close) {
473                 response.setHeader("Connection", "close");
474             } else {
475                 response.setHeader("Connection", "keep-alive");
476             }
477             response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
478             response.flushBuffer();
479         } catch (final IOException e) {
480             throw new RuntimeException(e);
481         }
482     }
