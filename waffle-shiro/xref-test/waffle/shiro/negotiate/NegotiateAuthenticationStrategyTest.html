24  package waffle.shiro.negotiate;
25  
26  import org.apache.shiro.realm.Realm;
27  import org.apache.shiro.realm.text.IniRealm;
28  import org.junit.jupiter.api.Assertions;
29  import org.junit.jupiter.api.BeforeEach;
30  import org.junit.jupiter.api.Test;
31  
32  /**
33   * The Class NegotiateAuthenticationStrategyTest.
34   *
35   * @author Dan Rollo Date: 3/18/13 Time: 3:34 PM
36   */
37  class NegotiateAuthenticationStrategyTest {
38  
39      /** The auth strategy. */
40      private NegotiateAuthenticationStrategy authStrategy;
41  
42      /**
43       * Sets the up.
44       */
45      @BeforeEach
46      void setUp() {
47          this.authStrategy = new NegotiateAuthenticationStrategy();
48      }
49  
50      /**
51       * Test after attempt.
52       */
53      @Test
54      void testAfterAttempt() {
55  
56          final Realm otherRealm = new IniRealm();
57  
58          this.authStrategy.afterAttempt(otherRealm, null, null, null, new RuntimeException());
59  
60          final AuthenticationInProgressException authInProgressException = new AuthenticationInProgressException();
61  
62          this.authStrategy.afterAttempt(otherRealm, null, null, null, authInProgressException);
63  
64          final NegotiateAuthenticationRealm realm = new NegotiateAuthenticationRealm();
65          final Throwable exception = Assertions.assertThrows(AuthenticationInProgressException.class, () -> {
66              this.authStrategy.afterAttempt(realm, null, null, null, authInProgressException);
67          });
68          Assertions.assertNull(exception.getMessage());
69      }
70  
71  }
