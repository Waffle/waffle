24  package waffle.shiro.negotiate;
25  
26  import mockit.Expectations;
27  import mockit.Mocked;
28  import mockit.Tested;
29  
30  import org.apache.shiro.authc.AuthenticationException;
31  import org.apache.shiro.authc.AuthenticationToken;
32  import org.junit.jupiter.api.Assertions;
33  import org.junit.jupiter.api.Test;
34  
35  /**
36   * The Class NegotiateAuthenticationRealmTest.
37   *
38   * @author Dan Rollo Date: 2/14/13 Time: 11:11 PM
39   */
40  class NegotiateAuthenticationRealmTest {
41  
42      /** The neg auth realm. */
43      @Tested
44      private NegotiateAuthenticationRealm negAuthRealm;
45  
46      @Mocked
47      AuthenticationToken authenticationToken;
48  
49      /**
50       * Test supports.
51       */
52      @Test
53      void testSupports() {
54          Assertions.assertFalse(this.negAuthRealm.supports(this.authenticationToken),
55                  "Non-NegotiateToken should not be supported.");
56  
57          Assertions
58                  .assertTrue(this.negAuthRealm.supports(new NegotiateToken(null, null, null, null, false, false, null)));
59      }
60  
61      /**
62       * Test authentication info exception.
63       *
64       * @param negotiateToken
65       *            the negotiate token
66       */
67      @Test
68      void testAuthenticationInfo(@Mocked final NegotiateToken negotiateToken) {
69          Assertions.assertNotNull(new Expectations() {
70              {
71                  negotiateToken.getIn();
72                  this.result = Byte.valueOf((byte) 0);
73              }
74          });
75          Assertions.assertThrows(AuthenticationException.class, () -> {
76              this.negAuthRealm.doGetAuthenticationInfo(negotiateToken);
77          });
78      }
79  }
