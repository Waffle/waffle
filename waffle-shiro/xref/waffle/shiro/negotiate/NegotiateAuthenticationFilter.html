123     @Override
124     protected AuthenticationToken createToken(final ServletRequest request, final ServletResponse response) {
125         final String authorization = this.getAuthzHeader(request);
126         final String[] elements = authorization.split(" ", -1);
127         final byte[] inToken = Base64.getDecoder().decode(elements[1]);
128 
129         // maintain a connection-based session for NTLM tokens
130         // TODO see about changing this parameter to ServletRequest in waffle
131         final String connectionId = NtlmServletRequest.getConnectionId((HttpServletRequest) request);
132         final String securityPackage = elements[0];
133 
134         // TODO see about changing this parameter to ServletRequest in waffle
135         final AuthorizationHeader authorizationHeader = new AuthorizationHeader((HttpServletRequest) request);
136         final boolean ntlmPost = authorizationHeader.isNtlmType1PostAuthorizationHeader();
137 
138         NegotiateAuthenticationFilter.LOGGER.debug("security package: {}, connection id: {}, ntlmPost: {}",
139                 securityPackage, connectionId, Boolean.valueOf(ntlmPost));
140 
141         final boolean rememberMe = this.isRememberMe(request);
142         final String host = this.getHost(request);
143 
144         return new NegotiateToken(inToken, new byte[0], connectionId, securityPackage, ntlmPost, rememberMe, host);
145     }

147     @Override
148     protected boolean onLoginSuccess(final AuthenticationToken token, final Subject subject,
149             final ServletRequest request, final ServletResponse response) throws Exception {
150         request.setAttribute("MY_SUBJECT", ((NegotiateToken) token).getSubject());
151         return true;
152     }
153 
154     @Override
155     protected boolean onLoginFailure(final AuthenticationToken token, final AuthenticationException e,
156             final ServletRequest request, final ServletResponse response) {
157         if (e instanceof AuthenticationInProgressException) {
158             // negotiate is processing
159             final String protocol = this.getAuthzHeaderProtocol(request);
160             NegotiateAuthenticationFilter.LOGGER.debug("Negotiation in progress for protocol: {}", protocol);
161             this.sendChallengeDuringNegotiate(protocol, response, ((NegotiateToken) token).getOut());
162             return false;
163         }
164         NegotiateAuthenticationFilter.LOGGER.warn("login exception: {}", e.getMessage());
165 
166         // do not send token.out bytes, this was a login failure.
167         this.sendChallengeOnFailure(response);
168 
169         this.setFailureAttribute(request, e);
170         return true;
171     }

205     @Override
206     protected boolean onAccessDenied(final ServletRequest request, final ServletResponse response) throws Exception {
207         // false by default or we wouldn't be in
208         boolean loggedIn = false;
209         // this method
210         if (this.isLoginAttempt(request)) {
211             loggedIn = this.executeLogin(request, response);
212         } else {
213             NegotiateAuthenticationFilter.LOGGER.debug("authorization required, supported protocols: {}",
214                     NegotiateAuthenticationFilter.PROTOCOLS);
215             this.sendChallengeInitiateNegotiate(response);
216         }
217         return loggedIn;
218     }

220     /**
221      * Determines whether the incoming request is an attempt to log in.
222      * <p/>
223      * The default implementation obtains the value of the request's
224      * {@link org.apache.shiro.web.filter.authc.BasicHttpAuthenticationFilter#AUTHORIZATION_HEADER AUTHORIZATION_HEADER}
225      * , and if it is not <code>null</code>, delegates to
226      * {@link org.apache.shiro.web.filter.authc.BasicHttpAuthenticationFilter#isLoginAttempt(String)
227      * isLoginAttempt(authzHeaderValue)}. If the header is <code>null</code>, <code>false</code> is returned.
228      *
229      * @param request
230      *            incoming ServletRequest
231      * @return true if the incoming request is an attempt to log in based, false otherwise
232      */
233     private boolean isLoginAttempt(final ServletRequest request) {
234         final String authzHeader = this.getAuthzHeader(request);
235         return authzHeader != null && this.isLoginAttempt(authzHeader);
236     }

238     /**
239      * Returns the {@link org.apache.shiro.web.filter.authc.BasicHttpAuthenticationFilter#AUTHORIZATION_HEADER
240      * AUTHORIZATION_HEADER} from the specified ServletRequest.
241      * <p/>
242      * This implementation merely casts the request to an <code>HttpServletRequest</code> and returns the header:
243      * <p/>
244      * <code>HttpServletRequest httpRequest = {@link WebUtils#toHttp(javax.servlet.ServletRequest) toHttp(reaquest)};<br/>
245      * return httpRequest.getHeader({@link org.apache.shiro.web.filter.authc.BasicHttpAuthenticationFilter#AUTHORIZATION_HEADER AUTHORIZATION_HEADER});</code>
246      *
247      * @param request
248      *            the incoming <code>ServletRequest</code>
249      * @return the <code>Authorization</code> header's value.
250      */
251     private String getAuthzHeader(final ServletRequest request) {
252         final HttpServletRequest httpRequest = WebUtils.toHttp(request);
253         return httpRequest.getHeader("Authorization");
254     }

256     /**
257      * Gets the authz header protocol.
258      *
259      * @param request
260      *            the request
261      * @return the authz header protocol
262      */
263     private String getAuthzHeaderProtocol(final ServletRequest request) {
264         final String authzHeader = this.getAuthzHeader(request);
265         return authzHeader.substring(0, authzHeader.indexOf(' '));
266     }

268     /**
269      * Default implementation that returns <code>true</code> if the specified <code>authzHeader</code> starts with the
270      * same (case-insensitive) characters specified by any of the configured protocols (Negotiate or NTLM),
271      * <code>false</code> otherwise.
272      *
273      * @param authzHeader
274      *            the 'Authorization' header value (guaranteed to be non-null if the
275      *            {@link #isLoginAttempt(javax.servlet.ServletRequest)} method is not overriden).
276      * @return <code>true</code> if the authzHeader value matches any of the configured protocols (Negotiate or NTLM).
277      */
278     boolean isLoginAttempt(final String authzHeader) {
279         for (final String protocol : NegotiateAuthenticationFilter.PROTOCOLS) {
280             if (authzHeader.toLowerCase().startsWith(protocol.toLowerCase())) {
281                 return true;
282             }
283         }
284         return false;
285     }

287     /**
288      * Builds the challenge for authorization by setting a HTTP <code>401</code> (Unauthorized) status as well as the
289      * response's {@link org.apache.shiro.web.filter.authc.BasicHttpAuthenticationFilter#AUTHENTICATE_HEADER
290      * AUTHENTICATE_HEADER}.
291      *
292      * @param protocols
293      *            protocols for which to send a challenge. In initial cases, will be all supported protocols. In the
294      *            midst of negotiation, will be only the protocol being negotiated.
295      *
296      * @param response
297      *            outgoing ServletResponse
298      * @param out
299      *            token.out or null
300      */
301     private void sendChallenge(final List<String> protocols, final ServletResponse response, final byte[] out) {
302         final HttpServletResponse httpResponse = WebUtils.toHttp(response);
303         this.sendAuthenticateHeader(protocols, out, httpResponse);
304         httpResponse.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
305     }

307     /**
308      * Send challenge initiate negotiate.
309      *
310      * @param response
311      *            the response
312      */
313     void sendChallengeInitiateNegotiate(final ServletResponse response) {
314         this.sendChallenge(NegotiateAuthenticationFilter.PROTOCOLS, response, null);
315     }

317     /**
318      * Send challenge during negotiate.
319      *
320      * @param protocol
321      *            the protocol
322      * @param response
323      *            the response
324      * @param out
325      *            the out
326      */
327     void sendChallengeDuringNegotiate(final String protocol, final ServletResponse response, final byte[] out) {
328         final List<String> protocolsList = new ArrayList<>();
329         protocolsList.add(protocol);
330         this.sendChallenge(protocolsList, response, out);
331     }
332 
333     /**
334      * Send challenge on failure.
335      *
336      * @param response
337      *            the response
338      */
339     void sendChallengeOnFailure(final ServletResponse response) {
340         final HttpServletResponse httpResponse = WebUtils.toHttp(response);
341         this.sendUnauthorized(NegotiateAuthenticationFilter.PROTOCOLS, null, httpResponse);
342         httpResponse.setHeader("Connection", "close");
343         try {
344             httpResponse.sendError(HttpServletResponse.SC_UNAUTHORIZED);
345             httpResponse.flushBuffer();
346         } catch (final IOException e) {
347             throw new RuntimeException(e);
348         }
349     }
350 
351     /**
352      * Send authenticate header.
353      *
354      * @param protocolsList
355      *            the protocols list
356      * @param out
357      *            the out
358      * @param httpResponse
359      *            the http response
360      */
361     private void sendAuthenticateHeader(final List<String> protocolsList, final byte[] out,
362             final HttpServletResponse httpResponse) {
363         this.sendUnauthorized(protocolsList, out, httpResponse);
364         httpResponse.setHeader("Connection", "keep-alive");
365     }
366 
367     /**
368      * Send unauthorized.
369      *
370      * @param protocols
371      *            the protocols
372      * @param out
373      *            the out
374      * @param response
375      *            the response
376      */
377     private void sendUnauthorized(final List<String> protocols, final byte[] out, final HttpServletResponse response) {
378         for (final String protocol : protocols) {
379             if (out == null || out.length == 0) {
380                 response.addHeader("WWW-Authenticate", protocol);
381             } else {
382                 response.setHeader("WWW-Authenticate", protocol + " " + Base64.getEncoder().encodeToString(out));
383             }
384         }
385     }
