24  package waffle.mock;
25  
26  import waffle.windows.auth.IWindowsAccount;
27  
28  /**
29   * The Class MockWindowsAccount.
30   */
31  public class MockWindowsAccount implements IWindowsAccount {
32  
33      /** The Constant TEST_USER_NAME. */
34      public static final String TEST_USER_NAME = "WaffleTestUser";
35  
36      /** The Constant TEST_PASSWORD. */
37      public static final String TEST_PASSWORD = "!WAFFLEP$$Wrd0";
38  
39      /** The fqn. */
40      private final String fqn;
41  
42      /** The name. */
43      private String name;
44  
45      /** The domain. */
46      private String domain;
47  
48      /** The sid. */
49      private final String sid;
50  
51      /**
52       * Instantiates a new mock windows account.
53       *
54       * @param newFqn
55       *            the new fqn
56       */
57      public MockWindowsAccount(final String newFqn) {
58          this(newFqn, "S-" + newFqn.hashCode());
59      }
60  
61      /**
62       * Instantiates a new mock windows account.
63       *
64       * @param newFqn
65       *            the new fqn
66       * @param newSid
67       *            the new sid
68       */
69      public MockWindowsAccount(final String newFqn, final String newSid) {
70          this.fqn = newFqn;
71          this.sid = newSid;
72          final String[] userNameDomain = newFqn.split("\\\\", 2);
73          if (userNameDomain.length == 2) {
74              this.name = userNameDomain[1];
75              this.domain = userNameDomain[0];
76          } else {
77              this.name = newFqn;
78          }
79      }
80  
81      @Override
82      public String getDomain() {
83          return this.domain;
84      }
85  
86      @Override
87      public String getFqn() {
88          return this.fqn;
89      }
90  
91      @Override
92      public String getName() {
93          return this.name;
94      }
95  
96      @Override
97      public String getSidString() {
98          return this.sid;
99      }
100 
101 }
