24  package waffle.util;
25  
26  import org.junit.jupiter.api.Assertions;
27  import org.junit.jupiter.api.Test;
28  
29  import waffle.mock.http.SimpleHttpRequest;
30  
31  /**
32   * The Class NtlmServletRequestTest.
33   */
34  class NtlmServletRequestTest {
35  
36      /**
37       * Test get connection id.
38       */
39      @Test
40      void testGetConnectionId() {
41          SimpleHttpRequest.resetRemotePort();
42          final SimpleHttpRequest request1 = new SimpleHttpRequest();
43          Assertions.assertEquals(":1", NtlmServletRequest.getConnectionId(request1));
44          final SimpleHttpRequest request2 = new SimpleHttpRequest();
45          Assertions.assertEquals(":2", NtlmServletRequest.getConnectionId(request2));
46          request2.setRemoteAddr("192.168.1.1");
47          Assertions.assertEquals("192.168.1.1:2", NtlmServletRequest.getConnectionId(request2));
48          request2.setRemoteHost("codeplex.com");
49          Assertions.assertEquals("codeplex.com:2", NtlmServletRequest.getConnectionId(request2));
50      }
51  }
