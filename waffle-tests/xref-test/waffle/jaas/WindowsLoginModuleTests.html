70      @Test
71      public void testLogin() throws LoginException {
72          Subject subject = new Subject();
73          UsernamePasswordCallbackHandler callbackHandler = new UsernamePasswordCallbackHandler(
74                  WindowsAccountImpl.getCurrentUsername(), "password");
75          Map<String, String> options = new HashMap<String, String>();
76          options.put("debug", "true");
77          this.loginModule.initialize(subject, callbackHandler, null, options);
78          assertTrue(this.loginModule.login());
79          assertEquals(0, subject.getPrincipals().size());
80          assertTrue(this.loginModule.commit());
81          assertEquals(3, subject.getPrincipals().size());
82          assertTrue(subject.getPrincipals().contains(new RolePrincipal("Everyone")));
83          assertTrue(subject.getPrincipals().contains(new RolePrincipal("Users")));
84          assertTrue(subject.getPrincipals().contains(new UserPrincipal(WindowsAccountImpl.getCurrentUsername())));
85          assertTrue(this.loginModule.logout());
86          assertSame(Integer.valueOf(subject.getPrincipals().size()), Integer.valueOf(0));
87      }

185     @Test(expected = LoginException.class)
186     public void testGuestLogin() throws LoginException {
187         Subject subject = new Subject();
188         UsernamePasswordCallbackHandler callbackHandler = new UsernamePasswordCallbackHandler("Guest", "password");
189         Map<String, String> options = new HashMap<String, String>();
190         options.put("debug", "true");
191         this.loginModule.initialize(subject, callbackHandler, null, options);
192         assertTrue(this.loginModule.isAllowGuestLogin());
193         assertTrue(this.loginModule.login());
194         assertEquals(0, subject.getPrincipals().size());
195         assertTrue(this.loginModule.commit());
196         assertEquals(3, subject.getPrincipals().size());
197         assertTrue(subject.getPrincipals().contains(new RolePrincipal("Everyone")));
198         assertTrue(subject.getPrincipals().contains(new RolePrincipal("Users")));
199         this.loginModule.setAllowGuestLogin(false);
200         assertTrue(this.loginModule.login());
201         fail("expected LoginException");
202     }
