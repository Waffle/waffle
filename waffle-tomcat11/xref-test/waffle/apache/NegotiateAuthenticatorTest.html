111      * Test challenge get.
112      */
113     @Test
114     void testChallengeGET() {
115         final SimpleHttpRequest request = new SimpleHttpRequest();
116         request.setMethod("GET");
117         final SimpleHttpResponse response = new SimpleHttpResponse(new Response());
118         this.authenticator.authenticate(request, response);
119         final String[] wwwAuthenticates = response.getHeaderValues("WWW-Authenticate");
120         Assertions.assertNotNull(wwwAuthenticates);
121         Assertions.assertEquals(2, wwwAuthenticates.length);
122         Assertions.assertEquals("Negotiate", wwwAuthenticates[0]);
123         Assertions.assertEquals("NTLM", wwwAuthenticates[1]);
124         Assertions.assertEquals("close", response.getHeader("Connection"));
125         Assertions.assertEquals(2, response.getHeaderNames().size());
126         Assertions.assertEquals(401, response.getStatus());
127     }

132     @Test
133     void testChallengePOST() {
134         final String securityPackage = "Negotiate";
135         IWindowsCredentialsHandle clientCredentials = null;
136         WindowsSecurityContextImpl clientContext = null;
137         try {
138             // client credentials handle
139             clientCredentials = WindowsCredentialsHandleImpl.getCurrent(securityPackage);
140             clientCredentials.initialize();
141             // initial client security context
142             clientContext = new WindowsSecurityContextImpl();
143             clientContext.setPrincipalName(WindowsAccountImpl.getCurrentUsername());
144             clientContext.setCredentialsHandle(clientCredentials);
145             clientContext.setSecurityPackage(securityPackage);
146             clientContext.initialize(null, null, WindowsAccountImpl.getCurrentUsername());
147             final SimpleHttpRequest request = new SimpleHttpRequest();
148             request.setMethod("POST");
149             request.setContentLength(0);
150             final String clientToken = Base64.getEncoder().encodeToString(clientContext.getToken());
151             request.addHeader("Authorization", securityPackage + " " + clientToken);
152             final SimpleHttpResponse response = new SimpleHttpResponse(new Response());
153             this.authenticator.authenticate(request, response);
154             Assertions.assertTrue(response.getHeader("WWW-Authenticate").startsWith(securityPackage + " "));
155             Assertions.assertEquals("keep-alive", response.getHeader("Connection"));
156             Assertions.assertEquals(2, response.getHeaderNames().size());
157             Assertions.assertEquals(401, response.getStatus());
158         } finally {
159             if (clientContext != null) {
160                 clientContext.dispose();
161             }
162             if (clientCredentials != null) {
163                 clientCredentials.dispose();
164             }
165         }

177     /**
178      * Test negotiate.
179      */
180     @Test
181     void testNegotiate() {
182         final String securityPackage = "Negotiate";
183         IWindowsCredentialsHandle clientCredentials = null;
184         WindowsSecurityContextImpl clientContext = null;
185         try {
186             // client credentials handle
187             clientCredentials = WindowsCredentialsHandleImpl.getCurrent(securityPackage);
188             clientCredentials.initialize();
189             // initial client security context
190             clientContext = new WindowsSecurityContextImpl();
191             clientContext.setPrincipalName(WindowsAccountImpl.getCurrentUsername());
192             clientContext.setCredentialsHandle(clientCredentials);
193             clientContext.setSecurityPackage(securityPackage);
194             clientContext.initialize(null, null, WindowsAccountImpl.getCurrentUsername());
195             // negotiate
196             boolean authenticated = false;
197             final SimpleHttpRequest request = new SimpleHttpRequest();
198             while (true) {
199                 final String clientToken = Base64.getEncoder().encodeToString(clientContext.getToken());
200                 request.addHeader("Authorization", securityPackage + " " + clientToken);
201 
202                 final SimpleHttpResponse response = new SimpleHttpResponse(new Response());
203                 authenticated = this.authenticator.authenticate(request, response);
204 
205                 if (authenticated) {
206                     Assertions.assertNotNull(request.getUserPrincipal());
207                     Assertions.assertTrue(request.getUserPrincipal() instanceof GenericWindowsPrincipal);
208                     final GenericWindowsPrincipal windowsPrincipal = (GenericWindowsPrincipal) request
209                             .getUserPrincipal();
210                     Assertions.assertTrue(windowsPrincipal.getSidString().startsWith("S-"));
211                     assertThat(windowsPrincipal.getSid()).isNotEmpty();
212                     Assertions.assertTrue(windowsPrincipal.getGroups().containsKey("Everyone"));
213                     assertThat(response.getHeaderNames()).hasSizeLessThanOrEqualTo(1);
214                     break;
215                 }
216 
217                 Assertions.assertTrue(response.getHeader("WWW-Authenticate").startsWith(securityPackage + " "));
218                 Assertions.assertEquals("keep-alive", response.getHeader("Connection"));
219                 Assertions.assertEquals(2, response.getHeaderNames().size());
220                 Assertions.assertEquals(401, response.getStatus());
221                 final String continueToken = response.getHeader("WWW-Authenticate")
222                         .substring(securityPackage.length() + 1);
223                 final byte[] continueTokenBytes = Base64.getDecoder().decode(continueToken);
224                 assertThat(continueTokenBytes).isNotEmpty();
225                 final ManagedSecBufferDesc continueTokenBuffer = new ManagedSecBufferDesc(Sspi.SECBUFFER_TOKEN,
226                         continueTokenBytes);
227                 clientContext.initialize(clientContext.getHandle(), continueTokenBuffer,
228                         WindowsAccountImpl.getCurrentUsername());
229             }
230             Assertions.assertTrue(authenticated);
231         } finally {
232             if (clientContext != null) {
233                 clientContext.dispose();
234             }
235             if (clientCredentials != null) {
236                 clientCredentials.dispose();
237             }
238         }
239     }

244     @Test
245     void testPOSTEmpty() {
246         final String securityPackage = "Negotiate";
247         IWindowsCredentialsHandle clientCredentials = null;
248         WindowsSecurityContextImpl clientContext = null;
249         try {
250             // client credentials handle
251             clientCredentials = WindowsCredentialsHandleImpl.getCurrent(securityPackage);
252             clientCredentials.initialize();
253             // initial client security context
254             clientContext = new WindowsSecurityContextImpl();
255             clientContext.setPrincipalName(WindowsAccountImpl.getCurrentUsername());
256             clientContext.setCredentialsHandle(clientCredentials);
257             clientContext.setSecurityPackage(securityPackage);
258             clientContext.initialize(null, null, WindowsAccountImpl.getCurrentUsername());
259             // negotiate
260             boolean authenticated = false;
261             final SimpleHttpRequest request = new SimpleHttpRequest();
262             request.setMethod("POST");
263             request.setContentLength(0);
264             String clientToken;
265             String continueToken;
266             byte[] continueTokenBytes;
267             SimpleHttpResponse response;
268             ManagedSecBufferDesc continueTokenBuffer;
269             while (true) {
270                 clientToken = Base64.getEncoder().encodeToString(clientContext.getToken());
271                 request.addHeader("Authorization", securityPackage + " " + clientToken);
272 
273                 response = new SimpleHttpResponse(new Response());
274                 authenticated = this.authenticator.authenticate(request, response);
275 
276                 if (authenticated) {
277                     assertThat(response.getHeaderNames().size()).isNotNegative();
278                     break;
279                 }
280 
281                 if (response.getHeader("WWW-Authenticate").startsWith(securityPackage + ",")) {
282                     Assertions.assertEquals("close", response.getHeader("Connection"));
283                     Assertions.assertEquals(2, response.getHeaderNames().size());
284                     Assertions.assertEquals(401, response.getStatus());
285                     return;
286                 }
287 
288                 Assertions.assertTrue(response.getHeader("WWW-Authenticate").startsWith(securityPackage + " "));
289                 Assertions.assertEquals("keep-alive", response.getHeader("Connection"));
290                 Assertions.assertEquals(2, response.getHeaderNames().size());
291                 Assertions.assertEquals(401, response.getStatus());
292                 continueToken = response.getHeader("WWW-Authenticate").substring(securityPackage.length() + 1);
293                 continueTokenBytes = Base64.getDecoder().decode(continueToken);
294                 assertThat(continueTokenBytes).isNotEmpty();
295                 continueTokenBuffer = new ManagedSecBufferDesc(Sspi.SECBUFFER_TOKEN, continueTokenBytes);
296                 clientContext.initialize(clientContext.getHandle(), continueTokenBuffer,
297                         WindowsAccountImpl.getCurrentUsername());
298             }
299             Assertions.assertTrue(authenticated);
300         } finally {
301             if (clientContext != null) {
302                 clientContext.dispose();
303             }
304             if (clientCredentials != null) {
305                 clientCredentials.dispose();
306             }
307         }
