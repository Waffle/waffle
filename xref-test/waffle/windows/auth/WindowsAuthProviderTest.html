80       * Test logon user.
81       */
82      @Test
83      void testLogonUser() {
84          final LMAccess.USER_INFO_1 userInfo = new LMAccess.USER_INFO_1();
85          userInfo.usri1_name = new WString("WaffleTestUser").toString();
86          userInfo.usri1_password = new WString("!WAFFLEP$$Wrd0").toString();
87          userInfo.usri1_priv = LMAccess.USER_PRIV_USER;
88          // ignore test if not able to add user (need to be administrator to do this).
89          Assumptions.assumeTrue(LMErr.NERR_Success == Netapi32.INSTANCE.NetUserAdd(null, 1, userInfo, null));
90          try {
91              final IWindowsAuthProvider prov = new WindowsAuthProviderImpl();
92              final IWindowsIdentity identity = prov.logonUser(userInfo.usri1_name.toString(),
93                      userInfo.usri1_password.toString());
94              Assertions.assertTrue(identity.getFqn().endsWith("\\" + userInfo.usri1_name.toString()));
95              Assertions.assertFalse(identity.isGuest());
96              identity.dispose();
97          } finally {
98              Assertions.assertEquals(LMErr.NERR_Success,
99                      Netapi32.INSTANCE.NetUserDel(null, userInfo.usri1_name.toString()));
100         }

103     /**
104      * Test impersonate logged on user.
105      */
106     @Test
107     void testImpersonateLoggedOnUser() {
108         final LMAccess.USER_INFO_1 userInfo = new LMAccess.USER_INFO_1();
109         userInfo.usri1_name = new WString(MockWindowsAccount.TEST_USER_NAME).toString();
110         userInfo.usri1_password = new WString(MockWindowsAccount.TEST_PASSWORD).toString();
111         userInfo.usri1_priv = LMAccess.USER_PRIV_USER;
112         // ignore test if not able to add user (need to be administrator to do this).
113         Assumptions.assumeTrue(LMErr.NERR_Success == Netapi32.INSTANCE.NetUserAdd(null, 1, userInfo, null));
114         try {
115             final IWindowsAuthProvider prov = new WindowsAuthProviderImpl();
116             final IWindowsIdentity identity = prov.logonUser(userInfo.usri1_name.toString(),
117                     userInfo.usri1_password.toString());
118             final IWindowsImpersonationContext ctx = identity.impersonate();
119             Assertions.assertTrue(userInfo.usri1_name.toString().equals(Advapi32Util.getUserName()));
120             ctx.revertToSelf();
121             Assertions.assertFalse(userInfo.usri1_name.toString().equals(Advapi32Util.getUserName()));
122             identity.dispose();
123         } finally {
124             Assertions.assertEquals(LMErr.NERR_Success,
125                     Netapi32.INSTANCE.NetUserDel(null, userInfo.usri1_name.toString()));
126         }

165     /**
166      * Test accept security token.
167      */
168     @Test
169     void testAcceptSecurityToken() {
170         final String securityPackage = "Negotiate";
171         final String targetName = "localhost";
172         IWindowsCredentialsHandle clientCredentials = null;
173         WindowsSecurityContextImpl clientContext = null;
174         IWindowsSecurityContext serverContext = null;
175         try {
176             // client credentials handle
177             clientCredentials = WindowsCredentialsHandleImpl.getCurrent(securityPackage);
178             clientCredentials.initialize();
179             // initial client security context
180             clientContext = new WindowsSecurityContextImpl();
181             clientContext.setPrincipalName(WindowsAccountImpl.getCurrentUsername());
182             clientContext.setCredentialsHandle(clientCredentials);
183             clientContext.setSecurityPackage(securityPackage);
184             clientContext.initialize(null, null, targetName);
185             // accept on the server
186             final WindowsAuthProviderImpl provider = new WindowsAuthProviderImpl();
187             final String connectionId = "testConnection-" + Thread.currentThread().getId();
188             do {
189                 // accept the token on the server
190                 serverContext = provider.acceptSecurityToken(connectionId, clientContext.getToken(), securityPackage);
191 
192                 if (serverContext != null && serverContext.isContinue()) {
193                     // initialize on the client
194                     final ManagedSecBufferDesc continueToken = new ManagedSecBufferDesc(Sspi.SECBUFFER_TOKEN,
195                             serverContext.getToken());
196                     clientContext.initialize(clientContext.getHandle(), continueToken, targetName);
197                     WindowsAuthProviderTest.LOGGER.info("Token: {}",
198                             Base64.getEncoder().encodeToString(serverContext.getToken()));
199                 }
200 
201             } while (serverContext != null && serverContext.isContinue());
202 
203             if (serverContext != null) {
204                 assertThat(serverContext.getIdentity().getFqn()).isNotEmpty();
205 
206                 WindowsAuthProviderTest.LOGGER.info(serverContext.getIdentity().getFqn());
207                 for (final IWindowsAccount group : serverContext.getIdentity().getGroups()) {
208                     WindowsAuthProviderTest.LOGGER.info(" {}", group.getFqn());
209                 }
210             }
211         } finally {
212             if (serverContext != null) {
213                 serverContext.dispose();
214             }
215             if (clientContext != null) {
216                 clientContext.dispose();
217             }
218             if (clientCredentials != null) {
219                 clientCredentials.dispose();
220             }
221         }
222     }

224     /**
225      * Test security contexts expire.
226      *
227      * @throws InterruptedException
228      *             the interrupted exception
229      */
230     @Test
231     void testSecurityContextsExpire() throws InterruptedException {
232         final String securityPackage = "Negotiate";
233         IWindowsCredentialsHandle clientCredentials = null;
234         WindowsSecurityContextImpl clientContext = null;
235         IWindowsSecurityContext serverContext = null;
236         try {
237             // client credentials handle
238             clientCredentials = WindowsCredentialsHandleImpl.getCurrent(securityPackage);
239             clientCredentials.initialize();
240             // initial client security context
241             clientContext = new WindowsSecurityContextImpl();
242             clientContext.setPrincipalName(WindowsAccountImpl.getCurrentUsername());
243             clientContext.setCredentialsHandle(clientCredentials);
244             clientContext.setSecurityPackage(securityPackage);
245             clientContext.initialize(null, null, WindowsAccountImpl.getCurrentUsername());
246             // accept on the server
247             final WindowsAuthProviderImpl provider = new WindowsAuthProviderImpl(1);
248             final int max = 100;
249             for (int i = 0; i < max; i++) {
250                 Thread.sleep(25);
251                 final String connectionId = "testConnection_" + i;
252                 serverContext = provider.acceptSecurityToken(connectionId, clientContext.getToken(), securityPackage);
253                 assertThat(provider.getContinueContextsSize()).isPositive();
254             }
255             WindowsAuthProviderTest.LOGGER.info("Cached security contexts: {}",
256                     Integer.valueOf(provider.getContinueContextsSize()));
257             Assertions.assertFalse(max == provider.getContinueContextsSize());
258         } finally {
259             if (serverContext != null) {
260                 serverContext.dispose();
261             }
262             if (clientContext != null) {
263                 clientContext.dispose();
264             }
265             if (clientCredentials != null) {
266                 clientCredentials.dispose();
267             }
268         }
269     }
