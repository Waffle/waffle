74      @Test
75      public void testFilter() {

77          assertEquals(PrincipalFormat.FQN, this.filter.getPrincipalFormat());
78          assertEquals(PrincipalFormat.BOTH, this.filter.getRoleFormat());
79          assertNull(this.filter.getFilterConfig());

83      @Test
84      public void testProvider() throws ClassNotFoundException {
85          final SecurityFilterProviderCollection provider = this.filter.getProvider();
86          assertEquals(2, provider.size());
87          assertTrue(provider.getByClassName("waffle.servlet.spi.BasicSecurityFilterProvider") instanceof BasicSecurityFilterProvider);
88          assertTrue(provider.getByClassName("waffle.servlet.spi.NegotiateSecurityFilterProvider") instanceof NegotiateSecurityFilterProvider);

91      @Test
92      public void testNoChallengeGET() throws IOException, ServletException {
93          final SimpleHttpRequest request = new SimpleHttpRequest();
94          request.setMethod("GET");
95          final SimpleHttpResponse response = new SimpleHttpResponse();
96          final SimpleFilterChain chain = new SimpleFilterChain();
97          this.filter.doFilter(request, response, chain);
98          // unlike servlet filters, it's a passthrough
99          assertEquals(500, response.getStatus());

102     @Test
103     public void testNegotiate() throws IOException, ServletException {
104         final String securityPackage = "Negotiate";
105         final SimpleFilterChain filterChain = new SimpleFilterChain();
106         final SimpleHttpRequest request = new SimpleHttpRequest();
107 
108         final String clientToken = BaseEncoding.base64().encode(
109                 WindowsAccountImpl.getCurrentUsername().getBytes(Charsets.UTF_8));
110         request.addHeader("Authorization", securityPackage + " " + clientToken);
111 
112         final SimpleHttpResponse response = new SimpleHttpResponse();
113         this.filter.doFilter(request, response, filterChain);
114 
115         final Authentication auth = SecurityContextHolder.getContext().getAuthentication();
116         assertNotNull(auth);
117         final Collection<? extends GrantedAuthority> authorities = auth.getAuthorities();
118         assertNotNull(authorities);
119         assertEquals(3, authorities.size());
120         final Iterator<? extends GrantedAuthority> authoritiesIterator = authorities.iterator();
121 
122         final List<String> list = new ArrayList<String>();
123         while (authoritiesIterator.hasNext()) {
124             list.add(authoritiesIterator.next().getAuthority());
125         }
126         Collections.sort(list);
127         assertEquals("ROLE_EVERYONE", list.get(0));
128         assertEquals("ROLE_USER", list.get(1));
129         assertEquals("ROLE_USERS", list.get(2));
130         assertEquals(0, response.getHeaderNamesSize());
131     }

133     @Test
134     public void testUnsupportedSecurityPackagePassthrough() throws IOException, ServletException {
135         final SimpleFilterChain filterChain = new SimpleFilterChain();
136         final SimpleHttpRequest request = new SimpleHttpRequest();
137         request.addHeader("Authorization", "Unsupported challenge");
138         final SimpleHttpResponse response = new SimpleHttpResponse();
139         this.filter.doFilter(request, response, filterChain);
140         // the filter should ignore authorization for an unsupported security package, ie. not return a 401
141         assertEquals(500, response.getStatus());
142     }

144     @Test
145     public void testGuestIsDisabled() throws IOException, ServletException {
146         final String securityPackage = "Negotiate";
147         final SimpleFilterChain filterChain = new SimpleFilterChain();
148         final SimpleHttpRequest request = new SimpleHttpRequest();
149 
150         final String clientToken = BaseEncoding.base64().encode("Guest".getBytes(Charsets.UTF_8));
151         request.addHeader("Authorization", securityPackage + " " + clientToken);
152 
153         final SimpleHttpResponse response = new SimpleHttpResponse();
154         this.filter.doFilter(request, response, filterChain);
155 
156         assertEquals(401, response.getStatus());
157         assertNull(SecurityContextHolder.getContext().getAuthentication());
158     }
