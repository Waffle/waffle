14  package waffle.util;
15  
16  import org.junit.Assert;
17  import org.junit.Test;
18  
19  /**
20   * The Class SPNegoMessageTests.
21   *
22   * @author ari.suutari[at]syncrontech[dot]com
23   */
24  public class SPNegoMessageTests {
25  
26      // Different SPNEGO messages. For details and specification,
27      // see http://msdn.microsoft.com/en-us/library/ms995330.aspx
28  
29      /** The Constant negTokenInitOk. */
30      private static final byte[] negTokenInitOk       = { 0x60, 0x76, 0x06, 0x06, 0x2B, 0x06, 0x01, 0x05, 0x05, 0x02 };
31  
32      /** The Constant negTokenInitTooShort. */
33      private static final byte[] negTokenInitTooShort = { 0x60, 0x76, 0x06, 0x06, 0x2B, 0x06, 0x01, 0x05, 0x05 };
34  
35      /** The Constant negTokenArgOk. */
36      private static final byte[] negTokenArgOk        = { (byte) 0xA1, 0x33, 0x30, 0x31, 0x0, 0x03, 0x0A, 0x01, 0x01,
37              0x0, 0x2A, 0x04, 0x28, 0x4E, 0x54, 0x4C, 0x4D, 0x53, 0x53, 0x50, 0x00, 0x01, 0x00, 0x00, 0x00, 0x0, 0x0,
38              0x08, 0x0, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00,
39              0x06, 0x01, 0x0, 0x1D, 0x00, 0x00, 0x00, 0x0F };
40  
41      /** The Constant negTokenArgTooShort. */
42      private static final byte[] negTokenArgTooShort  = { (byte) 0xA1, 0x33, 0x30, 0x31, 0x0, 0x03 };
43  
44      /** The Constant badMessage. */
45      private static final byte[] badMessage           = { 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00 };
46  
47      /**
48       * Test is sp nego message.
49       */
50      @Test
51      public void testIsSPNegoMessage() {
52          Assert.assertFalse(SPNegoMessage.isSPNegoMessage(null));
53          Assert.assertTrue(SPNegoMessage.isSPNegoMessage(SPNegoMessageTests.negTokenInitOk));
54          Assert.assertFalse(SPNegoMessage.isSPNegoMessage(SPNegoMessageTests.negTokenInitTooShort));
55          Assert.assertTrue(SPNegoMessage.isSPNegoMessage(SPNegoMessageTests.negTokenArgOk));
56          Assert.assertFalse(SPNegoMessage.isSPNegoMessage(SPNegoMessageTests.negTokenArgTooShort));
57          Assert.assertFalse(SPNegoMessage.isSPNegoMessage(SPNegoMessageTests.badMessage));
58      }
59  
60      /**
61       * Test is neg token init.
62       */
63      @Test
64      public void testIsNegTokenInit() {
65          Assert.assertTrue(SPNegoMessage.isNegTokenInit(SPNegoMessageTests.negTokenInitOk));
66          Assert.assertFalse(SPNegoMessage.isNegTokenInit(SPNegoMessageTests.negTokenInitTooShort));
67          Assert.assertFalse(SPNegoMessage.isNegTokenInit(SPNegoMessageTests.badMessage));
68      }
69  
70      /**
71       * Test is neg token arg.
72       */
73      @Test
74      public void testIsNegTokenArg() {
75          Assert.assertTrue(SPNegoMessage.isNegTokenArg(SPNegoMessageTests.negTokenArgOk));
76          Assert.assertFalse(SPNegoMessage.isNegTokenArg(SPNegoMessageTests.negTokenArgTooShort));
77          Assert.assertFalse(SPNegoMessage.isNegTokenArg(SPNegoMessageTests.badMessage));
78      }
79  }
