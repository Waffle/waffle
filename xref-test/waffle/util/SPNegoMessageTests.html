12  package waffle.util;
13  
14  import org.junit.jupiter.api.Assertions;
15  import org.junit.jupiter.api.Test;
16  
17  /**
18   * The Class SPNegoMessageTests.
19   *
20   * @author ari.suutari[at]syncrontech[dot]com
21   */
22  public class SPNegoMessageTests {
23  
24      // Different SPNEGO messages. For details and specification,
25      // see https://msdn.microsoft.com/en-us/library/ms995330.aspx
26  
27      /** The Constant negTokenInitOk. */
28      private static final byte[] negTokenInitOk = { 0x60, 0x76, 0x06, 0x06, 0x2B, 0x06, 0x01, 0x05, 0x05, 0x02 };
29  
30      /** The Constant negTokenInitTooShort. */
31      private static final byte[] negTokenInitTooShort = { 0x60, 0x76, 0x06, 0x06, 0x2B, 0x06, 0x01, 0x05, 0x05 };
32  
33      /** The Constant negTokenArgOk. */
34      private static final byte[] negTokenArgOk = { (byte) 0xA1, 0x33, 0x30, 0x31, 0x0, 0x03, 0x0A, 0x01, 0x01, 0x0, 0x2A,
35              0x04, 0x28, 0x4E, 0x54, 0x4C, 0x4D, 0x53, 0x53, 0x50, 0x00, 0x01, 0x00, 0x00, 0x00, 0x0, 0x0, 0x08, 0x0,
36              0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x06, 0x01,
37              0x0, 0x1D, 0x00, 0x00, 0x00, 0x0F };
38  
39      /** The Constant negTokenArgTooShort. */
40      private static final byte[] negTokenArgTooShort = { (byte) 0xA1, 0x33, 0x30, 0x31, 0x0, 0x03 };
41  
42      /** The Constant badMessage. */
43      private static final byte[] badMessage = { 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00 };
44  
45      /**
46       * Test is neg token init.
47       */
48      @Test
49      public void testIsNegTokenInit() {
50          Assertions.assertTrue(SPNegoMessage.isNegTokenInit(SPNegoMessageTests.negTokenInitOk));
51          Assertions.assertFalse(SPNegoMessage.isNegTokenInit(SPNegoMessageTests.negTokenInitTooShort));
52          Assertions.assertFalse(SPNegoMessage.isNegTokenInit(SPNegoMessageTests.badMessage));
53      }
54  
55      /**
56       * Test is neg token arg.
57       */
58      @Test
59      public void testIsNegTokenArg() {
60          Assertions.assertTrue(SPNegoMessage.isNegTokenArg(SPNegoMessageTests.negTokenArgOk));
61          Assertions.assertFalse(SPNegoMessage.isNegTokenArg(SPNegoMessageTests.negTokenArgTooShort));
62          Assertions.assertFalse(SPNegoMessage.isNegTokenArg(SPNegoMessageTests.badMessage));
63      }
64  }
