24  package waffle.util;
25  
26  import org.junit.jupiter.api.Assertions;
27  import org.junit.jupiter.api.Test;
28  
29  /**
30   * The Class NtlmMessageTest.
31   */
32  class NtlmMessageTest {
33  
34      /**
35       * Test is ntlm message.
36       */
37      @Test
38      void testIsNtlmMessage() {
39          Assertions.assertFalse(NtlmMessage.isNtlmMessage(null));
40          final byte[] ntlmSignature = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00 };
41          Assertions.assertTrue(NtlmMessage.isNtlmMessage(ntlmSignature));
42          final byte[] shortMessage = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50 };
43          Assertions.assertFalse(NtlmMessage.isNtlmMessage(shortMessage));
44          final byte[] longMessage = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x00 };
45          Assertions.assertTrue(NtlmMessage.isNtlmMessage(longMessage));
46          final byte[] badMessage = { 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00 };
47          Assertions.assertFalse(NtlmMessage.isNtlmMessage(badMessage));
48      }
49  
50      /**
51       * Test get ntlm message type.
52       */
53      @Test
54      void testGetNtlmMessageType() {
55          final byte[] ntlmMessageType1 = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x01, 0x00, 0x00, 0x00, 0x02,
56                  0x02, 0x00, 0x00 };
57          Assertions.assertEquals(1, NtlmMessage.getMessageType(ntlmMessageType1));
58          final byte[] ntlmMessageType2 = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x02, 0x00, 0x00, 0x00, 0x00,
59                  0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x02, 0x02, 0x00, 0x00, 0x01, 0x23, 0x45, 0x67 };
60          Assertions.assertEquals(2, NtlmMessage.getMessageType(ntlmMessageType2));
61          final byte[] ntlmMessageType3 = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x03, 0x00, 0x00, 0x00, 0x00,
62                  0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x02, 0x02, 0x00, 0x00, 0x01, 0x23, 0x45, 0x67 };
63          Assertions.assertEquals(3, NtlmMessage.getMessageType(ntlmMessageType3));
64      }
65  }
