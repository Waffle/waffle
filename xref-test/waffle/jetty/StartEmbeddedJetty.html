24  package waffle.jetty;
25  
26  import java.io.File;
27  import java.io.FileNotFoundException;
28  import java.io.IOException;
29  
30  import org.apache.jasper.servlet.JspServlet;
31  import org.eclipse.jetty.server.Server;
32  import org.eclipse.jetty.servlet.ServletHolder;
33  import org.eclipse.jetty.webapp.WebAppContext;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  
37  /**
38   * A simple embedded server that lets us run directly within Eclipse.
39   */
40  public class StartEmbeddedJetty {
41  
42      /** The Constant LOGGER. */
43      private static final Logger LOGGER = LoggerFactory.getLogger(StartEmbeddedJetty.class);
44  
45      /**
46       * Main method.
47       * 
48       * @param args
49       *            input arguments to main.
50       * @throws Exception
51       *             Exception thrown.
52       */
53      public static void main(final String[] args) throws Exception {
54          final String path = "../waffle-demo/waffle-filter";
55  
56          final File dir = new File(path);
57          if (!dir.exists()) {
58              throw new FileNotFoundException("Can not find webapp: " + dir.getAbsolutePath());
59          }
60  
61          final Server server = new Server(8080);
62          final WebAppContext context = new WebAppContext();
63          context.setServer(server);
64          context.setContextPath("/");
65          context.setWar(path);
66  
67          // Try adding JSP
68          final ServletHolder jsp = context.addServlet(JspServlet.class, "*.jsp");
69          jsp.setInitParameter("classpath", context.getClassPath());
70  
71          server.setHandler(context);
72  
73          try {
74              StartEmbeddedJetty.LOGGER.info(">>> STARTING EMBEDDED JETTY SERVER, PRESS ANY KEY TO STOP");
75              server.start();
76              if (System.in.read() == -1) {
77                  StartEmbeddedJetty.LOGGER.error("End of Stream reached");
78                  return;
79              }
80              StartEmbeddedJetty.LOGGER.info(">>> STOPPING EMBEDDED JETTY SERVER");
81              server.stop();
82              server.join();
83          } catch (final IOException e) {
84              StartEmbeddedJetty.LOGGER.error("", e);
85              System.exit(100);
86          } catch (final Exception e) {
87              StartEmbeddedJetty.LOGGER.error("", e);
88              System.exit(100);
89          }
90      }
91  
92  }
