14  package waffle.jetty;
15  
16  import java.io.File;
17  import java.io.FileNotFoundException;
18  import java.io.IOException;
19  
20  import org.apache.jasper.servlet.JspServlet;
21  import org.eclipse.jetty.server.Server;
22  import org.eclipse.jetty.servlet.ServletHolder;
23  import org.eclipse.jetty.webapp.WebAppContext;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  /**
28   * A simple embedded server that lets us run directly within Eclipse.
29   */
30  public class StartEmbeddedJetty {
31  
32      /** The Constant LOGGER. */
33      private static final Logger LOGGER = LoggerFactory.getLogger(StartEmbeddedJetty.class);
34  
35      /**
36       * Main method.
37       * 
38       * @param args
39       *            input arguments to main.
40       * @throws Exception
41       *             Exception thrown.
42       */
43      public static void main(final String[] args) throws Exception {
44          final String path = "../waffle-demo/waffle-filter";
45  
46          final File dir = new File(path);
47          if (!dir.exists()) {
48              throw new FileNotFoundException("Can not find webapp: " + dir.getAbsolutePath());
49          }
50  
51          final Server server = new Server(8080);
52          final WebAppContext context = new WebAppContext();
53          context.setServer(server);
54          context.setContextPath("/");
55          context.setWar(path);
56  
57          // Try adding JSP
58          final ServletHolder jsp = context.addServlet(JspServlet.class, "*.jsp");
59          jsp.setInitParameter("classpath", context.getClassPath());
60  
61          server.setHandler(context);
62  
63          try {
64              StartEmbeddedJetty.LOGGER.info(">>> STARTING EMBEDDED JETTY SERVER, PRESS ANY KEY TO STOP");
65              server.start();
66              System.in.read();
67              StartEmbeddedJetty.LOGGER.info(">>> STOPPING EMBEDDED JETTY SERVER");
68              server.stop();
69              server.join();
70          } catch (final IOException e) {
71              StartEmbeddedJetty.LOGGER.error("{}", e);
72              System.exit(100);
73          } catch (final Exception e) {
74              StartEmbeddedJetty.LOGGER.error("{}", e);
75              System.exit(100);
76          }
77      }
78  
79  }
