12  package waffle.apache.catalina;
13  
14  import java.security.Principal;
15  import java.util.HashMap;
16  import java.util.Map;
17  
18  import javax.servlet.http.HttpSession;
19  
20  import mockit.Mocked;
21  
22  import org.apache.catalina.connector.Request;
23  
24  /**
25   * Simple HTTP Request.
26   *
27   * @author dblock[at]dblock[dot]org
28   */
29  public class SimpleHttpRequest extends Request {
30  
31      /** The remote port s. */
32      private static int remotePortS;
33  
34      /**
35       * Next remote port.
36       *
37       * @return the int
38       */
39      public synchronized static int nextRemotePort() {
40          return ++SimpleHttpRequest.remotePortS;
41      }
42  
43      /**
44       * Reset remote port.
45       */
46      public synchronized static void resetRemotePort() {
47          SimpleHttpRequest.remotePortS = 0;
48      }
49  
50      /** The request uri. */
51      private String requestURI;
52  
53      /** The query string. */
54      private String queryString;
55  
56      /** The remote user. */
57      private String remoteUser;
58  
59      /** The method. */
60      private String method = "GET";
61  
62      /** The headers. */
63      private final Map<String, String> headers = new HashMap<>();
64  
65      /** The parameters. */
66      private final Map<String, String> parameters = new HashMap<>();
67  
68      /** The content. */
69      private byte[] content;
70  
71      /** The http session. */
72      @Mocked
73      private HttpSession httpSession;
74  
75      /** The principal. */
76      private Principal principal;
77  
78      /**
79       * Instantiates a new simple http request.
80       */
81      public SimpleHttpRequest() {
82          super();
83          this.remotePort = SimpleHttpRequest.nextRemotePort();
84      }
85  
86      /**
87       * Adds the header.
88       *
89       * @param headerName
90       *            the header name
91       * @param headerValue
92       *            the header value
93       */
94      public void addHeader(final String headerName, final String headerValue) {
95          this.headers.put(headerName, headerValue);
96      }
97  
98      /**
99       * Adds the parameter.
100      *
101      * @param parameterName
102      *            the parameter name
103      * @param parameterValue
104      *            the parameter value
105      */
106     public void addParameter(final String parameterName, final String parameterValue) {
107         this.parameters.put(parameterName, parameterValue);
108     }
109 
110     @Override
111     public int getContentLength() {
112         return this.content == null ? -1 : this.content.length;
113     }
114 
115     @Override
116     public String getHeader(final String headerName) {
117         return this.headers.get(headerName);
118     }
119 
120     @Override
121     public String getMethod() {
122         return this.method;
123     }
124 
125     @Override
126     public String getParameter(final String parameterName) {
127         return this.parameters.get(parameterName);
128     }
129 
130     @Override
131     public String getQueryString() {
132         return this.queryString;
133     }
134 
135     @Override
136     public String getRemoteAddr() {
137         return this.remoteAddr;
138     }
139 
140     @Override
141     public String getRemoteHost() {
142         return this.remoteHost;
143     }
144 
145     @Override
146     public int getRemotePort() {
147         return this.remotePort;
148     }
149 
150     @Override
151     public String getRemoteUser() {
152         return this.remoteUser;
153     }
154 
155     @Override
156     public String getRequestURI() {
157         return this.requestURI;
158     }
159 
160     @Override
161     public HttpSession getSession() {
162         return this.httpSession;
163     }
164 
165     @Override
166     public HttpSession getSession(final boolean create) {
167         return this.httpSession;
168     }
169 
170     @Override
171     public Principal getUserPrincipal() {
172         return this.principal;
173     }
174 
175     /**
176      * Sets the content length.
177      *
178      * @param length
179      *            the new content length
180      */
181     public void setContentLength(final int length) {
182         this.content = new byte[length];
183     }
184 
185     /**
186      * Sets the method.
187      *
188      * @param value
189      *            the new method
190      */
191     public void setMethod(final String value) {
192         this.method = value;
193     }
194 
195     /**
196      * Sets the query string.
197      *
198      * @param queryValue
199      *            the new query string
200      */
201     public void setQueryString(final String queryValue) {
202         this.queryString = queryValue;
203         if (this.queryString != null) {
204             for (final String eachParameter : this.queryString.split("[&]")) {
205                 final String[] pair = eachParameter.split("=");
206                 final String value = pair.length == 2 ? pair[1] : "";
207                 this.addParameter(pair[0], value);
208             }
209         }
210     }
211 
212     @Override
213     public void setRemoteAddr(final String value) {
214         this.remoteAddr = value;
215     }
216 
217     @Override
218     public void setRemoteHost(final String value) {
219         this.remoteHost = value;
220     }
221 
222     /**
223      * Sets the remote user.
224      *
225      * @param value
226      *            the new remote user
227      */
228     public void setRemoteUser(final String value) {
229         this.remoteUser = value;
230     }
231 
232     /**
233      * Sets the request uri.
234      *
235      * @param value
236      *            the new request uri
237      */
238     public void setRequestURI(final String value) {
239         this.requestURI = value;
240     }
241 
242     @Override
243     public void setUserPrincipal(final Principal value) {
244         this.principal = value;
245     }
246 
247 }
