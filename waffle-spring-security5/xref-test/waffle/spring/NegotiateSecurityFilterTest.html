90       */
91      @Test
92      void testFilter() {
93          Assertions.assertFalse(this.filter.isAllowGuestLogin());
94          Assertions.assertEquals(PrincipalFormat.FQN, this.filter.getPrincipalFormat());
95          Assertions.assertEquals(PrincipalFormat.BOTH, this.filter.getRoleFormat());
96          Assertions.assertNull(this.filter.getFilterConfig());
97          Assertions.assertNotNull(this.filter.getProvider());

106     @Test
107     void testProvider() throws ClassNotFoundException {
108         final SecurityFilterProviderCollection provider = this.filter.getProvider();

135     /**
136      * Test negotiate.
137      *
138      * @throws IOException
139      *             Signals that an I/O exception has occurred.
140      * @throws ServletException
141      *             the servlet exception
142      */
143     @Test
144     void testNegotiate() throws IOException, ServletException {
145         final String securityPackage = "Negotiate";
146         final SimpleFilterChain filterChain = new SimpleFilterChain();
147         final SimpleHttpRequest request = new SimpleHttpRequest();
148 
149         final String clientToken = Base64.getEncoder()
150                 .encodeToString(WindowsAccountImpl.getCurrentUsername().getBytes(StandardCharsets.UTF_8));
151         request.addHeader("Authorization", securityPackage + " " + clientToken);
152 
153         final SimpleHttpResponse response = new SimpleHttpResponse();
154         this.filter.doFilter(request, response, filterChain);
155 
156         final Authentication auth = SecurityContextHolder.getContext().getAuthentication();
157         Assertions.assertNotNull(auth);
158         final Collection<? extends GrantedAuthority> authorities = auth.getAuthorities();
159         Assertions.assertNotNull(authorities);
160         Assertions.assertEquals(3, authorities.size());
161 
162         final List<String> list = new ArrayList<>();
163         for (final GrantedAuthority grantedAuthority : authorities) {
164             list.add(grantedAuthority.getAuthority());
165         }
166         Collections.sort(list);
167         Assertions.assertEquals("ROLE_EVERYONE", list.get(0));
168         Assertions.assertEquals("ROLE_USER", list.get(1));
169         Assertions.assertEquals("ROLE_USERS", list.get(2));
170         Assertions.assertEquals(0, response.getHeaderNamesSize());
171     }

173     /**
174      * Test unsupported security package passthrough.
175      *
176      * @throws IOException
177      *             Signals that an I/O exception has occurred.
178      * @throws ServletException
179      *             the servlet exception
180      */
181     @Test
182     void testUnsupportedSecurityPackagePassthrough() throws IOException, ServletException {
183         final SimpleFilterChain filterChain = new SimpleFilterChain();
184         final SimpleHttpRequest request = new SimpleHttpRequest();
185         request.addHeader("Authorization", "Unsupported challenge");
186         final SimpleHttpResponse response = new SimpleHttpResponse();
187         this.filter.doFilter(request, response, filterChain);
188         // the filter should ignore authorization for an unsupported security package, ie. not return a 401
189         Assertions.assertEquals(500, response.getStatus());
190     }

192     /**
193      * Test guest is disabled.
194      *
195      * @throws IOException
196      *             Signals that an I/O exception has occurred.
197      * @throws ServletException
198      *             the servlet exception
199      */
200     @Test
201     void testGuestIsDisabled() throws IOException, ServletException {
202         final String securityPackage = "Negotiate";
203         final SimpleFilterChain filterChain = new SimpleFilterChain();
204         final SimpleHttpRequest request = new SimpleHttpRequest();
205 
206         final String clientToken = Base64.getEncoder().encodeToString("Guest".getBytes(StandardCharsets.UTF_8));
207         request.addHeader("Authorization", securityPackage + " " + clientToken);
208 
209         final SimpleHttpResponse response = new SimpleHttpResponse();
210         this.filter.doFilter(request, response, filterChain);
211 
212         Assertions.assertEquals(401, response.getStatus());
213         Assertions.assertNull(SecurityContextHolder.getContext().getAuthentication());
214     }
