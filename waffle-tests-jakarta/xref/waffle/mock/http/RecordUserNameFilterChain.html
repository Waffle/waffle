24  package waffle.mock.http;
25  
26  import com.sun.jna.platform.win32.Advapi32Util;
27  
28  import jakarta.servlet.ServletException;
29  import jakarta.servlet.ServletRequest;
30  import jakarta.servlet.ServletResponse;
31  
32  import java.io.IOException;
33  
34  /**
35   * Filter chain that records current username.
36   */
37  public class RecordUserNameFilterChain extends SimpleFilterChain {
38  
39      /** The user name. */
40      private String userName;
41  
42      @Override
43      public void doFilter(final ServletRequest servletRequest, final ServletResponse servletResponse)
44              throws IOException, ServletException {
45          super.doFilter(servletRequest, servletResponse);
46          this.userName = Advapi32Util.getUserName();
47      }
48  
49      /**
50       * Gets the user name.
51       *
52       * @return the user name
53       */
54      public String getUserName() {
55          return this.userName;
56      }
57  }
