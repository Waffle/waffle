188     /**
189      * Set the authentication protocols. Default is "Negotiate, NTLM".
190      *
191      * @param value
192      *            Authentication protocols
193      */
194     public void setProtocols(final String value) {
195         this.protocols = new LinkedHashSet<>();
196         final String[] protocolNames = value.split(",", -1);
197         for (String protocolName : protocolNames) {
198             protocolName = protocolName.trim();
199             if (!protocolName.isEmpty()) {
200                 this.log.debug("init protocol: {}", protocolName);
201                 if (WaffleAuthenticatorBase.SUPPORTED_PROTOCOLS.contains(protocolName)) {
202                     this.protocols.add(protocolName);
203                 } else {
204                     this.log.error("unsupported protocol: {}", protocolName);
205                     throw new RuntimeException("Unsupported protocol: " + protocolName);
206                 }
207             }
208         }
209     }
210 
211     /**
212      * Send a 401 Unauthorized along with protocol authentication headers.
213      *
214      * @param response
215      *            HTTP Response
216      */
217     protected void sendUnauthorized(final HttpServletResponse response) {
218         try {
219             for (final String protocol : this.protocols) {
220                 response.addHeader("WWW-Authenticate", protocol);
221             }
222             response.setHeader("Connection", "close");
223             response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
224             response.flushBuffer();
225         } catch (final IOException e) {
226             throw new RuntimeException(e);
227         }
228     }

241         } catch (final IOException e) {
242             this.log.error(e.getMessage());
243             this.log.trace("", e);
244             throw new RuntimeException(e);
245         }
246     }

253     @Override
254     protected Principal doLogin(final Request request, final String username, final String password)
255             throws ServletException {
256         this.log.debug("logging in: {}", username);
257         IWindowsIdentity windowsIdentity;
258         try {
259             windowsIdentity = this.auth.logonUser(username, password);
260         } catch (final Exception e) {
261             this.log.error(e.getMessage());
262             this.log.trace("", e);
263             return super.doLogin(request, username, password);
264         }
265         // disable guest login
266         if (!this.allowGuestLogin && windowsIdentity.isGuest()) {
267             this.log.warn("guest login disabled: {}", windowsIdentity.getFqn());
268             return super.doLogin(request, username, password);
269         }
270         try {
271             this.log.debug("successfully logged in {} ({})", username, windowsIdentity.getSidString());
272             final GenericPrincipal genericPrincipal = this.createPrincipal(windowsIdentity);
273             this.log.debug("roles: {}", String.join(", ", genericPrincipal.getRoles()));
274             return genericPrincipal;
275         } finally {
276             windowsIdentity.dispose();
277         }
278     }

280     /**
281      * This method will create an instance of a IWindowsIdentity based GenericPrincipal. It is used for creating custom
282      * implementation within subclasses.
283      *
284      * @param windowsIdentity
285      *            the windows identity to initialize the principal
286      * @return the Generic Principal
287      */
288     protected GenericPrincipal createPrincipal(final IWindowsIdentity windowsIdentity) {
289         return new GenericWindowsPrincipal(windowsIdentity, this.principalFormat, this.roleFormat);
290     }

303         super.startInternal();
