77      private GrantedAuthorityFactory grantedAuthorityFactory = WindowsAuthenticationToken.DEFAULT_GRANTED_AUTHORITY_FACTORY;

82      /**
83       * Instantiates a new negotiate security filter.
84       */
85      public NegotiateSecurityFilter() {
86          super();
87          NegotiateSecurityFilter.LOGGER.debug("[waffle.spring.NegotiateSecurityFilter] loaded");
88      }
89  
90      @Override
91      public void doFilter(final ServletRequest req, final ServletResponse res, final FilterChain chain)
92              throws IOException, ServletException {
93  
94          final HttpServletRequest request = (HttpServletRequest) req;
95          final HttpServletResponse response = (HttpServletResponse) res;
96  
97          NegotiateSecurityFilter.LOGGER.debug("{} {}, contentlength: {}", request.getMethod(), request.getRequestURI(),
98                  Integer.valueOf(request.getContentLength()));
99  
100         final AuthorizationHeader authorizationHeader = new AuthorizationHeader(request);
101 
102         // authenticate user
103         if (!authorizationHeader.isNull()
104                 && this.provider.isSecurityPackageSupported(authorizationHeader.getSecurityPackage())) {
105 
106             // log the user in using the token
107             IWindowsIdentity windowsIdentity;
108 
109             try {
110                 windowsIdentity = this.provider.doFilter(request, response);
111                 if (windowsIdentity == null) {
112                     return;
113                 }
114             } catch (final IOException e) {
115                 NegotiateSecurityFilter.LOGGER.warn("error logging in user: {}", e.getMessage());
116                 NegotiateSecurityFilter.LOGGER.trace("", e);
117                 this.sendUnauthorized(response, true);
118                 return;
119             }
120 
121             if (!this.allowGuestLogin && windowsIdentity.isGuest()) {
122                 NegotiateSecurityFilter.LOGGER.warn("guest login disabled: {}", windowsIdentity.getFqn());
123                 this.sendUnauthorized(response, true);
124                 return;
125             }
126 
127             IWindowsImpersonationContext ctx = null;
128             try {
129                 NegotiateSecurityFilter.LOGGER.debug("logged in user: {} ({})", windowsIdentity.getFqn(),
130                         windowsIdentity.getSidString());
131 
132                 final WindowsPrincipal principal = this.impersonate
133                         ? new AutoDisposableWindowsPrincipal(windowsIdentity, this.principalFormat, this.roleFormat)
134                         : new WindowsPrincipal(windowsIdentity, this.principalFormat, this.roleFormat);
135 
136                 NegotiateSecurityFilter.LOGGER.debug("roles: {}", principal.getRolesString());
137 
138                 final Authentication authentication = new WindowsAuthenticationToken(principal,
139                         this.grantedAuthorityFactory, this.defaultGrantedAuthority);
140 
141                 if (!this.setAuthentication(request, response, authentication)) {
142                     return;
143                 }
144 
145                 NegotiateSecurityFilter.LOGGER.info("successfully logged in user: {}", windowsIdentity.getFqn());
146 
147                 if (this.impersonate) {
148                     NegotiateSecurityFilter.LOGGER.debug("impersonating user");
149                     ctx = windowsIdentity.impersonate();
150                 }
151 
152                 chain.doFilter(request, response);
153             } finally {
154                 if (this.impersonate && ctx != null) {
155                     NegotiateSecurityFilter.LOGGER.debug("terminating impersonation");
156                     ctx.revertToSelf();
157                 } else {
158                     windowsIdentity.dispose();
159                 }
160             }
161         } else {
162             chain.doFilter(request, response);
163         }
164     }

166     /**
167      * Invoked when authentication towards ad was succesful to populate securitycontext Override to add service provider
168      * authorization checks.
169      *
170      * @param request
171      *            the request
172      * @param response
173      *            the response
174      * @param authentication
175      *            the authentication
176      * @return true, if successful
177      */
178     protected boolean setAuthentication(final HttpServletRequest request, final HttpServletResponse response,
179             final Authentication authentication) {
180         SecurityContextHolder.getContext().setAuthentication(authentication);
181         return true;
182     }
183 
184     @Override
185     public void afterPropertiesSet() throws ServletException {
186         super.afterPropertiesSet();
187 
188         if (this.provider == null) {
189             throw new ServletException("Missing NegotiateSecurityFilter.Provider");
190         }
191     }

193     /**
194      * Send a 401 Unauthorized along with protocol authentication headers.
195      *
196      * @param response
197      *            HTTP Response
198      * @param close
199      *            Close connection.
200      */
201     protected void sendUnauthorized(final HttpServletResponse response, final boolean close) {
202         try {
203             this.provider.sendUnauthorized(response);
204             if (close) {
205                 response.setHeader("Connection", "close");
206             } else {
207                 response.setHeader("Connection", "keep-alive");
208             }
209             response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
210             response.flushBuffer();
211         } catch (final IOException e) {
212             throw new RuntimeException(e);
213         }
214     }
