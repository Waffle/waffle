12  package waffle.spring.handlers;

14  import java.io.IOException;
15  
16  import javax.servlet.ServletException;
17  import javax.servlet.http.HttpServletRequest;
18  import javax.servlet.http.HttpServletResponse;
19  
20  import org.springframework.security.access.AccessDeniedException;
21  import org.springframework.security.web.access.AccessDeniedHandler;
22  
23  public class CustomAccessDeniedHandler implements AccessDeniedHandler {
24  
25      @Override
26      public void handle(final HttpServletRequest request, final HttpServletResponse response,
27              final AccessDeniedException ex) throws IOException, ServletException {
28          // here we can add custom headers or response status, data or redirect
29          response.addHeader("Custom-header", "some value");
30          response.sendError(HttpServletResponse.SC_UNAUTHORIZED);
31  
32      }
33  
34  }
