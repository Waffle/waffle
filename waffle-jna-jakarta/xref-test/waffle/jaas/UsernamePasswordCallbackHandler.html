24  package waffle.jaas;
25  
26  import java.io.IOException;
27  
28  import javax.security.auth.callback.Callback;
29  import javax.security.auth.callback.CallbackHandler;
30  import javax.security.auth.callback.NameCallback;
31  import javax.security.auth.callback.PasswordCallback;
32  import javax.security.auth.callback.UnsupportedCallbackException;
33  
34  /**
35   * The Class UsernamePasswordCallbackHandler.
36   */
37  public class UsernamePasswordCallbackHandler implements CallbackHandler {
38  
39      /** The username. */
40      private final String username;
41  
42      /** The password. */
43      private final String password;
44  
45      /**
46       * Instantiates a new username password callback handler.
47       *
48       * @param newUserName
49       *            the new user name
50       * @param newPassword
51       *            the new password
52       */
53      public UsernamePasswordCallbackHandler(final String newUserName, final String newPassword) {
54          this.username = newUserName;
55          this.password = newPassword;
56      }
57  
58      @Override
59      public void handle(final Callback[] cb) throws IOException, UnsupportedCallbackException {
60          for (final Callback element : cb) {
61              if (element instanceof NameCallback) {
62                  final NameCallback nc = (NameCallback) element;
63                  nc.setName(this.username);
64              } else if (element instanceof PasswordCallback) {
65                  final PasswordCallback pc = (PasswordCallback) element;
66                  pc.setPassword(this.password == null ? null : this.password.toCharArray());
67              } else {
68                  throw new UnsupportedCallbackException(element, "UsernamePasswordCallbackHandler");
69              }
70          }
71      }
72  
73  }
