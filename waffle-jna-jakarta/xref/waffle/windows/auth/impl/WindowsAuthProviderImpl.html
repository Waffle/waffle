61      /** The Continue Context Timeout. */
62      public static final int CONTINUE_CONTEXT_TIMEOUT = 30;
63  
64      /**
65       * The Class ContinueContext.
66       */
67      private static class ContinueContext {
68          /** The continue handle. */
69          CtxtHandle continueHandle;
70  
71          /** The server credential. */
72          IWindowsCredentialsHandle serverCredential;
73  
74          /**
75           * Instantiates a new continue context.
76           *
77           * @param handle
78           *            the handle
79           * @param windowsCredential
80           *            the windows credential
81           */
82          public ContinueContext(final CtxtHandle handle, final IWindowsCredentialsHandle windowsCredential) {
83              this.continueHandle = handle;
84              this.serverCredential = windowsCredential;
85          }
86      }
87  
88      /** The continue contexts. */
89      private final Cache<String, ContinueContext> continueContexts;
90  
91      /**
92       * Instantiates a new windows auth provider impl.
93       */
94      public WindowsAuthProviderImpl() {
95          this(WindowsAuthProviderImpl.CONTINUE_CONTEXT_TIMEOUT);
96      }

98      /**
99       * A Windows authentication provider.
100      *
101      * @param continueContextsTimeout
102      *            Timeout for security contexts in seconds.
103      */
104     public WindowsAuthProviderImpl(final int continueContextsTimeout) {
105         this.continueContexts = Cache.newCache(continueContextsTimeout);
106     }

108     @Override
109     public IWindowsSecurityContext acceptSecurityToken(final String connectionId, final byte[] token,
110             final String securityPackage) {
111 
112         if (token == null || token.length == 0) {
113             this.resetSecurityToken(connectionId);
114             throw new Win32Exception(WinError.SEC_E_INVALID_TOKEN);
115         }
116 
117         CtxtHandle continueHandle = null;
118         IWindowsCredentialsHandle serverCredential;
119         ContinueContext continueContext = this.continueContexts.get(connectionId);
120         if (continueContext != null) {
121             continueHandle = continueContext.continueHandle;
122             serverCredential = continueContext.serverCredential;
123         } else {
124             serverCredential = new WindowsCredentialsHandleImpl(null, Sspi.SECPKG_CRED_INBOUND, securityPackage);
125             serverCredential.initialize();
126         }
127 
128         WindowsSecurityContextImpl sc;
129 
130         int rc;
131         int tokenSize = Sspi.MAX_TOKEN_SIZE;
132 
133         do {
134             final ManagedSecBufferDesc pbServerToken = new ManagedSecBufferDesc(Sspi.SECBUFFER_TOKEN, tokenSize);
135             final ManagedSecBufferDesc pbClientToken = new ManagedSecBufferDesc(Sspi.SECBUFFER_TOKEN, token);
136             final IntByReference pfClientContextAttr = new IntByReference();
137 
138             final CtxtHandle phNewServerContext = new CtxtHandle();
139             rc = Secur32.INSTANCE.AcceptSecurityContext(serverCredential.getHandle(), continueHandle, pbClientToken,
140                     Sspi.ISC_REQ_CONNECTION, Sspi.SECURITY_NATIVE_DREP, phNewServerContext, pbServerToken,
141                     pfClientContextAttr, null);
142 
143             sc = new WindowsSecurityContextImpl();
144             sc.setCredentialsHandle(serverCredential);
145             sc.setSecurityPackage(securityPackage);
146             sc.setSecurityContext(phNewServerContext);
147 
148             switch (rc) {
149                 case WinError.SEC_E_BUFFER_TOO_SMALL:
150                     tokenSize += Sspi.MAX_TOKEN_SIZE;
151                     sc.dispose();
152                     WindowsSecurityContextImpl.dispose(continueHandle);
153                     break;
154                 case WinError.SEC_E_OK:
155                     // the security context received from the client was accepted
156                     this.resetSecurityToken(connectionId);
157                     // if an output token was generated by the function, it must be sent to the client process
158                     if (pbServerToken.pBuffers != null && pbServerToken.cBuffers == 1
159                             && pbServerToken.getBuffer(0).cbBuffer > 0) {
160                         sc.setToken(pbServerToken.getBuffer(0).getBytes() == null ? new byte[0]
161                                 : pbServerToken.getBuffer(0).getBytes().clone());
162                     }
163                     sc.setContinue(false);
164                     break;
165                 case WinError.SEC_I_CONTINUE_NEEDED:
166                     // the server must send the output token to the client and wait for a returned token
167                     continueContext = new ContinueContext(phNewServerContext, serverCredential);
168                     this.continueContexts.put(connectionId, continueContext);
169                     sc.setToken(pbServerToken.getBuffer(0).getBytes() == null ? new byte[0]
170                             : pbServerToken.getBuffer(0).getBytes().clone());
171                     sc.setContinue(true);
172                     break;
173                 default:
174                     sc.dispose();
175                     WindowsSecurityContextImpl.dispose(continueHandle);
176                     this.resetSecurityToken(connectionId);
177                     throw new Win32Exception(rc);
178             }
179         } while (rc == WinError.SEC_E_BUFFER_TOO_SMALL);
180 
181         return sc;
182     }

184     @Override
185     public IWindowsComputer getCurrentComputer() {
186         try {
187             return new WindowsComputerImpl(InetAddress.getLocalHost().getHostName());
188         } catch (final UnknownHostException e) {
189             throw new RuntimeException(e);
190         }
191     }
192 
193     @Override
194     public IWindowsDomain[] getDomains() {
195         final List<IWindowsDomain> domains = new ArrayList<>();
196         final DomainTrust[] trusts = Netapi32Util.getDomainTrusts();
197         for (final DomainTrust trust : trusts) {
198             domains.add(new WindowsDomainImpl(trust));
199         }
200         return domains.toArray(new IWindowsDomain[0]);
201     }
202 
203     @Override
204     public IWindowsIdentity logonDomainUser(final String username, final String domain, final String password) {
205         return this.logonDomainUserEx(username, domain, password, WinBase.LOGON32_LOGON_NETWORK,
206                 WinBase.LOGON32_PROVIDER_DEFAULT);
207     }
208 
209     @Override
210     public IWindowsIdentity logonDomainUserEx(final String username, final String domain, final String password,
211             final int logonType, final int logonProvider) {
212         final HANDLEByReference phUser = new HANDLEByReference();
213         if (!Advapi32.INSTANCE.LogonUser(username, domain, password, logonType, logonProvider, phUser)) {
214             throw new Win32Exception(Kernel32.INSTANCE.GetLastError());
215         }
216         return new WindowsIdentityImpl(phUser.getValue());
217     }
218 
219     @Override
220     public IWindowsIdentity logonUser(final String username, final String password) {
221         // username@domain UPN format is natively supported by the
222         // Windows LogonUser API process domain\\username format
223         final String[] userNameDomain = username.split("\\\\", 2);
224         if (userNameDomain.length == 2) {
225             return this.logonDomainUser(userNameDomain[1], userNameDomain[0], password);
226         }
227         return this.logonDomainUser(username, null, password);
228     }
229 
230     @Override
231     public IWindowsAccount lookupAccount(final String username) {
232         return new WindowsAccountImpl(username);
233     }
234 
235     @Override
236     public void resetSecurityToken(final String connectionId) {
237         this.continueContexts.remove(connectionId);
238     }

242      *
243      * @return Number of elements in the hash map.
244      */
245     public int getContinueContextsSize() {
246         return this.continueContexts.size();
247     }
