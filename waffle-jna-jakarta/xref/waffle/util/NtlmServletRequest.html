24  package waffle.util;
25  
26  import jakarta.servlet.http.HttpServletRequest;
27  
28  /**
29   * The Class NtlmServletRequest.
30   */
31  public final class NtlmServletRequest {
32  
33      /**
34       * Instantiates a new ntlm servlet request.
35       */
36      private NtlmServletRequest() {
37          // Prevent Instantiation of object
38      }
39  
40      /**
41       * Returns a unique connection id for a given servlet request.
42       *
43       * @param request
44       *            Servlet request.
45       *
46       * @return String.
47       */
48      public static String getConnectionId(final HttpServletRequest request) {
49          final String remoteHost = NtlmServletRequest.getRemoteHost(request);
50          return String.join(":", remoteHost == null ? "" : remoteHost, String.valueOf(request.getRemotePort()));
51      }
52  
53      /**
54       * Gets the remote host.
55       *
56       * @param request
57       *            the request
58       *
59       * @return the remote host
60       */
61      private static String getRemoteHost(final HttpServletRequest request) {
62          return request.getRemoteHost() == null ? request.getRemoteAddr() : request.getRemoteHost();
63      }
64  
65  }
