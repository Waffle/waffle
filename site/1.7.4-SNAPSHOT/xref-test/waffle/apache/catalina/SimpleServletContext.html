14  package waffle.apache.catalina;
15  
16  import javax.servlet.RequestDispatcher;
17  import javax.servlet.ServletContext;
18  
19  import org.mockito.Mockito;
20  
21  /**
22   * Simple Servlet Context.
23   * 
24   * @author dblock[at]dblock[dot]org
25   */
26  public abstract class SimpleServletContext implements ServletContext {
27  
28      /**
29       * Get Request Dispatcher used by Waffle.
30       */
31      @Override
32      public RequestDispatcher getRequestDispatcher(final String url) {
33          final SimpleRequestDispatcher dispatcher = Mockito.mock(SimpleRequestDispatcher.class,
34                  Mockito.CALLS_REAL_METHODS);
35          dispatcher.setUrl(url);
36          return dispatcher;
37      }
38  
39  }
